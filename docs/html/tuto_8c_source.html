35static int tutorial_parse_numeric_png(const char* file_name, int* out_number) {
36 if (!file_name || !out_number) return EXIT_FAILURE;
37
38 size_t len = strlen(file_name);
39 if (len < 5) return EXIT_FAILURE;
40 if (strcmp(file_name + (len - 4), ".png") != 0) return EXIT_FAILURE;
41
42 size_t number_len = len - 4;
43 if (number_len == 0 || number_len >= 32) return EXIT_FAILURE;
44
45 char number_buf[32] = {0};
46 for (size_t i = 0; i < number_len; i++) {
47 unsigned char ch = (unsigned char)file_name[i];
48 if (!isdigit(ch)) return EXIT_FAILURE;
49 number_buf[i] = (char)ch;
50 }
51
52 *out_number = atoi(number_buf);
53 return EXIT_SUCCESS;
54}

79static int tutorial_collect_files(TutorialImageFile** out_files, int* out_count) {
80 if (!out_files || !out_count) return EXIT_FAILURE;
81 *out_files = NULL;
82 *out_count = 0;
83
84#ifdef _WIN32
85 WIN32_FIND_DATAA find_data;
86 HANDLE h_find = FindFirstFileA(TUTO_SLIDES_GLOB, &find_data);
87 if (h_find == INVALID_HANDLE_VALUE) {
88 return EXIT_SUCCESS;
89 }
90
91 do {
92 if (find_data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) continue;
93
94 int number = 0;
95 if (tutorial_parse_numeric_png(find_data.cFileName, &number) != EXIT_SUCCESS) continue;
96
97 if (tutorial_append_file(out_files, out_count, number, find_data.cFileName) != EXIT_SUCCESS) {
98 FindClose(h_find);
99 return EXIT_FAILURE;
100 }
101 } while (FindNextFileA(h_find, &find_data));
102
103 FindClose(h_find);
104#else
105 DIR* dir = opendir(TUTO_SLIDES_DIR);
106 if (!dir) return EXIT_SUCCESS;
107
108 struct dirent* ent = NULL;
109 while ((ent = readdir(dir)) != NULL) {
110 int number = 0;
111 if (tutorial_parse_numeric_png(ent->d_name, &number) != EXIT_SUCCESS) continue;
112
113 if (tutorial_append_file(out_files, out_count, number, ent->d_name) != EXIT_SUCCESS) {
114 closedir(dir);
115 return EXIT_FAILURE;
116 }
117 }
118
119 closedir(dir);
120#endif
121
122 if (*out_count > 1) {
123 qsort(*out_files, (size_t)*out_count, sizeof(TutorialImageFile), tutorial_cmp_file);
124 }
125
126 return EXIT_SUCCESS;
127}

183 g_tuto.slides = resized ? resized : slides;

309static void tutorial_render_slide(SDL_Renderer* renderer) {
310 if (!renderer || !g_tuto.window || !g_tuto.window->cfg) return;
311
312 SDL_Rect content = {0};
313 if (tutorial_get_content_rect(&content) != EXIT_SUCCESS) return;
314
315 SDL_SetRenderDrawBlendMode(renderer, SDL_BLENDMODE_BLEND);
316 SDL_SetRenderDrawColor(renderer, 15, 15, 15, 220);
317 SDL_RenderFillRect(renderer, &content);
318 SDL_SetRenderDrawColor(renderer, 90, 90, 90, 255);
319 SDL_RenderDrawRect(renderer, &content);
320
321 SDL_Texture* slide = tutorial_current_slide();
322 if (!slide) return;
323
324 int tex_w = 0;
325 int tex_h = 0;
326 if (SDL_QueryTexture(slide, NULL, NULL, &tex_w, &tex_h) != 0) return;
327 if (tex_w <= 0 || tex_h <= 0) return;
328
329 double sx = (double)content.w / (double)tex_w;
330 double sy = (double)content.h / (double)tex_h;
331 double scale = (sx < sy) ? sx : sy;
332 if (scale <= 0.0) return;
333
334 SDL_Rect dst = {0};
335 dst.w = (int)lround((double)tex_w * scale);
336 dst.h = (int)lround((double)tex_h * scale);
337 dst.x = content.x + (content.w - dst.w) / 2;
338 dst.y = content.y + (content.h - dst.h) / 2;
339
340 SDL_RenderSetClipRect(renderer, &content);
341 SDL_RenderCopy(renderer, slide, NULL, &dst);
342 SDL_RenderSetClipRect(renderer, NULL);
343}
