179/* ratio=0 -> rouge, ratio=1 -> vert */
180static SDL_Color color_red_to_green(float ratio) {
181 float t = clamp01(ratio);
182 Uint8 r = (Uint8)(255.0f * (1.0f - t));
183 Uint8 g = (Uint8)(255.0f * t);
184 return (SDL_Color){r, g, 0, 255};
185}

187static SDL_Color color_for_fps(float fps) {
188 const float fps_min = 30.0f;
189 const float fps_max = 60.0f;
190 float ratio = (fps - fps_min) / (fps_max - fps_min);
191 return color_red_to_green(ratio);
192}

194static SDL_Color color_for_ping(int ping_ms) {

227void infos_display(SDL_Context* context) {
228 if (bandeau) {
229 // Récupérer la position de la souris
230 int mouse_x, mouse_y;
231 SDL_GetMouseState(&mouse_x, &mouse_y);
232 int logical_mouse_x = mouse_x;
233 int logical_mouse_y = mouse_y;
234 window_to_logical(context, mouse_x, mouse_y, &logical_mouse_x, &logical_mouse_y);
235
236 // Constantes pour le positionnement
237 const int TRIGGER_X = 300; // Seuil X pour déclencher l'animation
238 const int BASE_X = -950; // Position visible du bandeau
239 const int HIDDEN_X = -1300; // Position hors écran (caché)
240
241 // Vérifier si la souris est dedans (au-dessus du seuil)
242 int mouse_inside = (logical_mouse_x <= TRIGGER_X);
243
244 // Déterminer l'état actuel et les positions d'animation
245 Uint32 current_time = SDL_GetTicks();
246 float progress = 0.0f;
247 int display_x = HIDDEN_X;
248
249 // Gérer les transitions d'état basées sur la souris
250 if (mouse_inside && !mouse_was_inside) {
251 // La souris vient d'entrer : déclencher l'apparition
252 infos_state = INFOS_SHOWING;
253 animation_start_time = SDL_GetTicks();
254 mouse_was_inside = 1;
255 }
256 else if (!mouse_inside && mouse_was_inside) {
257 // La souris vient de sortir : déclencher la disparition
258 infos_state = INFOS_HIDING;
259 animation_start_time = SDL_GetTicks();
260 mouse_was_inside = 0;
261 }
262
263 // Gérer les animations en cours
264 if (infos_state == INFOS_SHOWING) {
265 progress = (float)(current_time - animation_start_time) / ANIMATION_DURATION;
266 if (progress >= 1.0f) {
267 progress = 1.0f;
268 infos_state = INFOS_VISIBLE;
269 }
270 // Interpolation du sliding en apparition : HIDDEN_X -> BASE_X
271 display_x = HIDDEN_X + (BASE_X - HIDDEN_X) * progress;
272 }
273 else if (infos_state == INFOS_HIDING) {
274 progress = (float)(current_time - animation_start_time) / ANIMATION_DURATION;
275 if (progress >= 1.0f) {
276 progress = 1.0f;
277 infos_state = INFOS_HIDDEN;
278 }
279 // Interpolation du sliding en disparition : BASE_X -> HIDDEN_X
280 display_x = BASE_X + (HIDDEN_X - BASE_X) * progress;
281 }
282 else if (infos_state == INFOS_VISIBLE) {
283 // Reste visible à la position finale
284 display_x = BASE_X;
285 }
286 // Si INFOS_HIDDEN, display_x reste HIDDEN_X
287
288 current_display_x = display_x;
289 update_crossfader_positions(display_x);
290
291 // N'afficher que si pas complètement caché
292 if (infos_state != INFOS_HIDDEN) {
293 display_image(context->renderer, bandeau, display_x, 0, 0.75, 90, SDL_FLIP_NONE, 1, 192);
294 fps_ping_display(context, display_x);
295
296 /* Labels des crossfaders (coordonnées bandeau, même repère que text_display) */
297 int label_x = display_x + 100;
298 text_display(context->renderer, "Musique", FONT_LARABIE, 18, COL_WHITE, label_x, 345, 0, 255);
299 text_display(context->renderer, "Effets Sonores", FONT_LARABIE, 18, COL_WHITE, label_x, 275, 0, 255);
300
301 /* Affichage des règles du jeu */
302 int rules_start_y = 180; // Position Y de départ (en dessous des volumes)
303 int line_spacing = 22; // Espacement entre les lignes
304
305 // Titre des règles
306 text_display(context->renderer, rules_title, FONT_LARABIE, 20, COL_WHITE, label_x, rules_start_y, 0, 255);
307
308 // Lignes des règles
309 for (int i = 0; i < rules_line_count; i++) {
310 text_display(context->renderer, rules_lines[i], FONT_LARABIE, 14, COL_WHITE, label_x, rules_start_y - 35 - (i * line_spacing), 0, 200);
311 }
312
313 /* Rendu des crossfaders */
314 crossfaders_render(context->renderer);
315 }
316 }

323// FPS calculation
324void calculate_fps(SDL_Context* context, Uint32 current_time) {
325 static Uint32 last_time = 0;
326 static int frame_count = 0;
327
328 if (last_time == 0) {
329 last_time = current_time;
330 }
331
332 frame_count++;
333 if (current_time - last_time >= 100) { // Mettre à jour les FPS toutes les 100 ms
334 context->fps = (frame_count * 1000.0f) / (current_time - last_time);
335 frame_count = 0;
336 last_time = current_time;
337 }
338}
