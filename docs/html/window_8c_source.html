172void window_handle_event(AppContext* ctx, Window* win, SDL_Event* event) {
173 if (!ctx || !win || !win->cfg || !event) return;
174
175 WindowConfig* cfg = win->cfg;
176 if (cfg->hidden) return;
177
178 if (event->type == SDL_MOUSEBUTTONDOWN && event->button.button == SDL_BUTTON_LEFT) {
179 int mx = event->button.x;
180 int my = event->button.y;
181
182 /*
183 * Autoriser le drag depuis n'importe quelle zone de la fenêtre
184 * (pas seulement la barre de titre) sauf si un champ de saisie
185 * texte est actif globalement (SDL_IsTextInputActive()).
186 * Cela évite d'interférer avec la saisie dans les inputs.
187 */
188 if (cfg->movable && (point_in_rect(mx, my, &cfg->rect) && !SDL_IsTextInputActive())) {
189 cfg->dragging = 1;
190 cfg->drag_offset_x = mx - cfg->rect.x;
191 cfg->drag_offset_y = my - cfg->rect.y;
192 }
193 } else if (event->type == SDL_MOUSEMOTION) {
194 if (!cfg->dragging) return;
195
196 int mx = event->motion.x;
197 int my = event->motion.y;
198
199 int new_rect_x = mx - cfg->drag_offset_x;
200 int new_rect_y = my - cfg->drag_offset_y;
201
202 cfg->x = new_rect_x + (cfg->w / 2) - (WIN_WIDTH / 2);
203 cfg->y = (WIN_HEIGHT / 2) - (new_rect_y + (cfg->h / 2));
204 window_update_rect(win);
205 } else if (event->type == SDL_MOUSEBUTTONUP && event->button.button == SDL_BUTTON_LEFT) {
206 cfg->dragging = 0;
207 }
208}

210void window_render(SDL_Renderer* renderer, const Window* win) {

222 }
223
224 if (cfg->titlebar_h > 0) {
225 SDL_Rect titlebar = {
226 .x = cfg->rect.x,
227 .y = cfg->rect.y,
228 .w = cfg->rect.w,
229 .h = cfg->titlebar_h
230 };
231 if (cfg->titlebar_texture) {
232 SDL_RenderCopy(renderer, cfg->titlebar_texture, NULL, &titlebar);

235 SDL_RenderFillRect(renderer, &titlebar);
236 }
237 window_render_title(renderer, cfg);
238 }
239

242 SDL_SetRenderDrawColor(renderer, cfg->border_color.r, cfg->border_color.g, cfg->border_color.b, cfg->border_color.a);
243 for (int i = 0; i < border; i++) {
244 SDL_Rect r = {
245 .x = cfg->rect.x + i,
246 .y = cfg->rect.y + i,
247 .w = cfg->rect.w - (2 * i),
248 .h = cfg->rect.h - (2 * i)
249 };
250 if (r.w <= 0 || r.h <= 0) break;
251 SDL_RenderDrawRect(renderer, &r);
252 }

376 if (!renderer || !win || !texture) return EXIT_FAILURE;
377
378 int og_w = 0;
379 int og_h = 0;
380 if (SDL_QueryTexture(texture, NULL, NULL, &og_w, &og_h) != 0) return EXIT_FAILURE;
381
382 if (size_factor <= 0.0f) return EXIT_FAILURE;
383
384 int base_w = (ratio != 1.0f) ? (int)(og_w * ratio) : og_w;
385 if (base_w <= 0) return EXIT_FAILURE;
386
387 int draw_w = (int)(base_w * size_factor);
388 int draw_h = (int)(og_h * size_factor);
389 if (draw_w <= 0 || draw_h <= 0) return EXIT_FAILURE;
390
391 int origin_x = 0;
392 int origin_y = 0;
393 if (window_content_origin_screen(win, &origin_x, &origin_y) != EXIT_SUCCESS) return EXIT_FAILURE;

395 /* rel_x/rel_y: coordonnées relatives au centre de la fenêtre
396 * rel_x positif = droite, rel_y positif = haut
397 * L'image sera centrée sur cette position */
398 int screen_x = origin_x + rel_x;
399 int screen_y = origin_y - rel_y;
400
401 /* Convertir en coordonnées logiques pour display_image */
402 int logical_x = screen_x - (WIN_WIDTH / 2);
403 int logical_y = (WIN_HEIGHT / 2) - screen_y;
404
