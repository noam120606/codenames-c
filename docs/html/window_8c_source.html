100void window_handle_event(AppContext* ctx, Window* win, SDL_Event* event) {
101 if (!ctx || !win || !win->cfg || !event) return;
102
103 WindowConfig* cfg = win->cfg;
104 if (cfg->hidden) return;
105
106 if (event->type == SDL_MOUSEBUTTONDOWN && event->button.button == SDL_BUTTON_LEFT) {
107 int mx = event->button.x;
108 int my = event->button.y;
109
110 if (cfg->movable && point_in_titlebar(win, mx, my)) {
111 cfg->dragging = 1;
112 cfg->drag_offset_x = mx - cfg->rect.x;
113 cfg->drag_offset_y = my - cfg->rect.y;
114 }
115 } else if (event->type == SDL_MOUSEMOTION) {
116 if (!cfg->dragging) return;
117
118 int mx = event->motion.x;
119 int my = event->motion.y;
120
121 int new_rect_x = mx - cfg->drag_offset_x;
122 int new_rect_y = my - cfg->drag_offset_y;
123
124 cfg->x = new_rect_x + (cfg->w / 2) - (WIN_WIDTH / 2);
125 cfg->y = (WIN_HEIGHT / 2) - (new_rect_y + (cfg->h / 2));
126 window_update_rect(win);
127 } else if (event->type == SDL_MOUSEBUTTONUP && event->button.button == SDL_BUTTON_LEFT) {
128 cfg->dragging = 0;
129 }
130}

152

154 SDL_SetRenderDrawColor(renderer, cfg->border_color.r, cfg->border_color.g, cfg->border_color.b, cfg->border_color.a);
155 for (int i = 0; i < border; i++) {
156 SDL_Rect r = {
157 .x = cfg->rect.x + i,
158 .y = cfg->rect.y + i,
159 .w = cfg->rect.w - (2 * i),
160 .h = cfg->rect.h - (2 * i)
161 };
162 if (r.w <= 0 || r.h <= 0) break;
163 SDL_RenderDrawRect(renderer, &r);

186 case WIN_CFG_DRAGGING: cfg->dragging = ((int)value != 0); break;
