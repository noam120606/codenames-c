45static int window_scrollable_zone_rect_from_cfg(const WindowConfig* cfg, SDL_Rect* out_rect) {
46 if (!cfg || !out_rect || !cfg->scrollable) return EXIT_FAILURE;
47
48 int center_x = cfg->rect.x + (cfg->rect.w / 2);
49 int center_y = cfg->rect.y + (cfg->rect.h / 2);
50
51 /* x vers la droite, y vers le haut (repère local fenêtre) */
52 int p1_x = center_x + cfg->scroll_x1;
53 int p1_y = center_y - cfg->scroll_y1;
54 int p2_x = center_x + cfg->scroll_x2;
55 int p2_y = center_y - cfg->scroll_y2;
56
57 int left = (p1_x < p2_x) ? p1_x : p2_x;
58 int right = (p1_x > p2_x) ? p1_x : p2_x;
59 int top = (p1_y < p2_y) ? p1_y : p2_y;
60 int bottom = (p1_y > p2_y) ? p1_y : p2_y;
61
62 out_rect->x = left;
63 out_rect->y = top;
64 out_rect->w = right - left;
65 out_rect->h = bottom - top;

312void window_handle_event(AppContext* ctx, Window* win, SDL_Event* event) {
313 if (!ctx || !win || !win->cfg || !event) return;
314
315 WindowConfig* cfg = win->cfg;
316 if (cfg->hidden) return;
317
318 if (event->type == SDL_MOUSEBUTTONDOWN && event->button.button == SDL_BUTTON_LEFT) {
319 int mx = event->button.x;
320 int my = event->button.y;
321
322 /*
323 * Autoriser le drag depuis n'importe quelle zone de la fenêtre
324 * (pas seulement la barre de titre) sauf si un champ de saisie
325 * texte est actif globalement (SDL_IsTextInputActive()).
326 * Cela évite d'interférer avec la saisie dans les inputs.
327 */
328 if (cfg->movable && (point_in_rect(mx, my, &cfg->rect) && !SDL_IsTextInputActive())) {
329 cfg->dragging = 1;
330 cfg->drag_offset_x = mx - cfg->rect.x;
331 cfg->drag_offset_y = my - cfg->rect.y;
332 }
333 } else if (event->type == SDL_MOUSEMOTION) {
334 if (!cfg->dragging) return;
335
336 int mx = event->motion.x;
337 int my = event->motion.y;
338
339 int new_rect_x = mx - cfg->drag_offset_x;
340 int new_rect_y = my - cfg->drag_offset_y;
341
342 cfg->x = new_rect_x + (cfg->w / 2) - (WIN_WIDTH / 2);
343 cfg->y = (WIN_HEIGHT / 2) - (new_rect_y + (cfg->h / 2));
344 window_update_rect(win);
345 } else if (event->type == SDL_MOUSEWHEEL) {
346 if (!cfg->scrollable) return;
347
348 int raw_mx = 0;
349 int raw_my = 0;
350 SDL_GetMouseState(&raw_mx, &raw_my);
351
352 int mx = raw_mx;
353 int my = raw_my;
354 window_to_logical_coords(ctx, raw_mx, raw_my, &mx, &my);
355
356 SDL_Rect zone = {0};
357 if (window_scrollable_zone_rect_from_cfg(cfg, &zone) != EXIT_SUCCESS) return;
358 if (!point_in_rect(mx, my, &zone)) return;
359
360 int wheel_y = event->wheel.y;
361 if (event->wheel.direction == SDL_MOUSEWHEEL_FLIPPED) {
362 wheel_y = -wheel_y;
363 }
364 if (wheel_y == 0) return;
365
366 cfg->scroll_offset += wheel_y * cfg->scroll_step;
367 window_normalize_scroll_cfg(cfg);
368 } else if (event->type == SDL_MOUSEBUTTONUP && event->button.button == SDL_BUTTON_LEFT) {
369 cfg->dragging = 0;
370 }
371}
