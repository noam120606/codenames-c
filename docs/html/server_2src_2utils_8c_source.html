6int randint(int min, int max) {
7 // Retourne un entier aléatoire entre min et max inclus
8 // Formule : min + rand() / (RAND_MAX / (max - min + 1) + 1)
9 int range = max - min + 1;
10 return min + rand() / (RAND_MAX / range + 1);
