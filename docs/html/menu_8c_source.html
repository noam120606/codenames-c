115static float menu_ease_out_back(float t) {
116 const float c1 = MENU_BOUNCE_OVERSHOOT;
117 const float c3 = c1 + 1.0f;
118 float x = t - 1.0f;
119 return 1.0f + c3 * x * x * x + c1 * x * x;
120}

135static Uint8 menu_opening_credits_alpha(Uint32 elapsed_ms, Uint32 delay_ms) {
136 if (elapsed_ms <= delay_ms) return 0;
137
138 if (MENU_STARTUP_CREATORS_FADE_MS == 0) return 255;
139
140 Uint32 delayed_elapsed_ms = elapsed_ms - delay_ms;
141
142 if (delayed_elapsed_ms < MENU_STARTUP_CREATORS_FADE_MS) {
143 float fade_in_t = (float)delayed_elapsed_ms / (float)MENU_STARTUP_CREATORS_FADE_MS;
144 return (Uint8)(255.0f * menu_clamp01(fade_in_t));
145 }
146
147 if (elapsed_ms >= MENU_STARTUP_OPENING_MS) {
148 return 0;
149 }
150
151 if (elapsed_ms > (MENU_STARTUP_OPENING_MS - MENU_STARTUP_CREATORS_FADE_MS)) {
152 Uint32 remaining_ms = MENU_STARTUP_OPENING_MS - elapsed_ms;
153 float fade_out_t = (float)remaining_ms / (float)MENU_STARTUP_CREATORS_FADE_MS;
154 return (Uint8)(255.0f * menu_clamp01(fade_out_t));
155 }
156
157 return 255;
158}

452static void menu_update_startup_state(AppContext* context) {
453 if (!context) return;
454
455 Uint32 now = SDL_GetTicks();
456
457 if (menu_startup.phase == MENU_STARTUP_PHASE_LOADING) {
458 float logo_progress = menu_clamp01((float)(now - menu_startup.startup_started_at_ms) / (float)MENU_STARTUP_LOGO_FADE_MS);
459 if (menu_startup.loading_complete && logo_progress >= 1.0f) {
460 if (menu_startup.skip_requested) {
461 menu_enter_startup_transition(now);
462 } else {
463 menu_startup.phase = MENU_STARTUP_PHASE_OPENING;
464 menu_startup.opening_started_at_ms = now;
465 menu_startup.opening_audio_started = 0;
466 }
467 }
468 }
469
470 if (menu_startup.phase == MENU_STARTUP_PHASE_OPENING) {
471 if (!menu_startup.opening_audio_started) {
472 (void)audio_play_with_fade(SOUND_OPENING_CODENAMES, 0, 150, AUDIO_FADE_IN_BY_VOLUME, NULL);
473 menu_startup.opening_audio_started = 1;
474 }
475
476 if (menu_startup.skip_requested || (now - menu_startup.opening_started_at_ms) >= MENU_STARTUP_OPENING_MS) {
477 menu_enter_startup_transition(now);
478 }
479 }
480
481 if (menu_startup.phase == MENU_STARTUP_PHASE_TRANSITION && !menu_startup.transition_initialized) {
482 menu_capture_ui_targets();
483 menu_prepare_bounce_starts();
484 menu_startup.transition_initialized = 1;
485 }
