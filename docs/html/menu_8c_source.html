168 /* Compter les lignes */
169 int line_count = 0;

171 while (fgets(line_buf, sizeof(line_buf), uf)) line_count++;
172
173 if (line_count > 0) {
174 srand((unsigned int)SDL_GetTicks());
175 int chosen = rand() % line_count;
176
177 rewind(uf);
178 int cur = 0;
179 while (fgets(line_buf, sizeof(line_buf), uf)) {
180 if (cur == chosen) {
181 /* Retirer le '\n' final */
182 size_t len = strlen(line_buf);
183 if (len > 0 && line_buf[len - 1] == '\n') line_buf[len - 1] = '\0';

190 input_submit(context, name_input);
191 }
192 break;
193 }
194 cur++;
195 }
