101 else if ((c & 0xF8) == 0xF0) char_len = 4;

104 char_len = 1;

109 candidate_len = HISTORY_LINE_SIZE - 1;

114 probe[candidate_len] = '\0';

135 unsigned char first = (unsigned char)cursor[0];
136 if ((first & 0x80) == 0x00) split_at = 1;
137 else if ((first & 0xE0) == 0xC0) split_at = 2;
138 else if ((first & 0xF0) == 0xE0) split_at = 3;
139 else if ((first & 0xF8) == 0xF0) split_at = 4;

143 if (split_at > remaining_len) split_at = remaining_len;
