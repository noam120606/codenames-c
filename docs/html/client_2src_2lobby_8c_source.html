82static void lobby_sync_local_user_in_roster(AppContext* context) {
83 if (!context || !context->lobby || context->player_id < 0) return;
84
85 const char* local_name = (context->player_name && context->player_name[0] != '\0') ? context->player_name : "Unknown";
86 int slot = lobby_find_user_slot_by_id(context->lobby, context->player_id);
87
88 if (slot >= 0) {
89 User* local_user = context->lobby->users[slot];
90 if (!local_user) return;
91
92 if (!local_user->name || strcmp(local_user->name, local_name) != 0) {
93 char* copy = strdup(local_name);
94 if (!copy) return;
95 free(local_user->name);
96 local_user->name = copy;
97 }
98
99 local_user->role = context->player_role;
100 local_user->team = context->player_team;
101 return;
102 }
103
104 int free_slot = lobby_find_free_user_slot(context->lobby);
105 if (free_slot < 0) return;
106
107 User* created = create_user(context->player_id, local_name, context->player_role, context->player_team);
108 if (!created) return;
109
110 context->lobby->users[free_slot] = created;
111 context->lobby->nb_players++;
112}

114static void count_player_for_layout(const User* user, PlayerPlacementCounters* counters) {
115 if (!user || !counters) return;
116
117 const char* name = user->name ? user->name : "Unknown";
118 if (user->role == ROLE_NONE) {
119 counters->nb_none++;
120 return;
121 }
122
123 switch (user->team) {
124 case TEAM_RED:
125 user->role == ROLE_SPY ? counters->nb_red_spy++ : counters->nb_red_agent++;
126 break;
127 case TEAM_BLUE:
128 user->role == ROLE_SPY ? counters->nb_blue_spy++ : counters->nb_blue_agent++;
129 break;
130 case TEAM_NONE:
131 counters->nb_none++;
132 break;
133 default:
134 printf("Warning: User %s has invalid team %d\n", name, user->team);
135 break;
136 }

164static int compute_player_icon_position(int nb_player, int i_player, int base_x, int base_y, int* out_x, int* out_y) {
165 if (!out_x || !out_y) return 0;
166
167 const int spacing_x = 80;
168 const int spacing_y = 80;
169
170 if (nb_player < 1 || nb_player > 8) {
171 printf("Too many players for lobby display: %d\n", nb_player);
172 return 0;
173 }
174
175 if (i_player < 0 || i_player >= nb_player) {
176 return 0;
177 }
178
179 int first_line_count = 0;
180 switch (nb_player) {
181 case 1: first_line_count = 1; break;
182 case 2: first_line_count = 2; break;
183 case 3: first_line_count = 3; break;
184 case 4: first_line_count = 4; break;
185 case 5: first_line_count = 3; break;
186 case 6: first_line_count = 3; break;
187 case 7: first_line_count = 4; break;
188 case 8: first_line_count = 4; break;
189 default: return 0;
190 }
191
192 int second_line_count = nb_player - first_line_count;
193
194 int line_count = 0;
195 int line_index = 0;
196 int is_second_line = (i_player >= first_line_count);
197
198 if (is_second_line) {
199 line_count = second_line_count;
200 line_index = i_player - first_line_count;
201 *out_y = base_y - spacing_y;
202 } else {
203 line_count = first_line_count;
204 line_index = i_player;
205 *out_y = base_y;
206 }
207
208 switch (line_count) {
209 case 1:
210 *out_x = base_x;
211 break;
212 case 2:
213 if (is_second_line) {
214 *out_x = base_x + ((line_index == 0) ? -spacing_x/2 : spacing_x/2); // Pour les 2 joueurs de la deuxième ligne : cas avec 5 joueurs.
215 } else {
216 *out_x = base_x + ((line_index == 0) ? -spacing_x : spacing_x);
217 }
218 break;
219 case 3:
220 *out_x = base_x + (line_index - 1) * spacing_x;
221 break;
222 case 4:
223 *out_x = base_x + (line_index * spacing_x) - (int)(1.5f * spacing_x);
224 break;
225 default:
226 return 0;
227 }
228
229 return 1;
230}

425 /* Informer le serveur du lancement de la partie (si protocole / message existant) */
426 char msg[16];
427 format_to(msg, sizeof(msg), "%d", MSG_STARTGAME);
428 send_tcp(context->sock, msg);
429 printf("Launch game requested\n");
430 } else if (button == btn_return) {

434 char msg[16];
435 format_to(msg, sizeof(msg), "%d", MSG_LEAVELOBBY);

437 }

439 context->player_team = TEAM_NONE;
440 context->lobby->id = -1;
441
442 /* Retirer le filtre audio en quittant le lobby. */
443 audio_set_filter(MUSIC_MENU_LOBBY, AUDIO_FILTER_NONE, 0);
444 context->app_state = APP_STATE_MENU;
445 printf("Returned to menu\n");
446 } else if (button == btn_words_difficulty_switch) {
447 // Alterne entre normale et difficile
448 switch (context->lobby->words_difficulty) {
449 case WORDS_DIFFICULTY_NORMAL:
450 context->lobby->words_difficulty = WORDS_DIFFICULTY_HARD;
451 break;
452 case WORDS_DIFFICULTY_HARD:
453 context->lobby->words_difficulty = WORDS_DIFFICULTY_INFO;
454 break;
455 case WORDS_DIFFICULTY_INFO:
456 if (word_contains("67", context->player_name)) {
457 context->lobby->words_difficulty = WORDS_DIFFICULTY_FREAKY;
458 } else {
459 context->lobby->words_difficulty = WORDS_DIFFICULTY_NORMAL;
460 }
461 break;
462 default:
463 context->lobby->words_difficulty = WORDS_DIFFICULTY_NORMAL;
464 break;
465 }
466 char msg[16];
467 format_to(msg, sizeof(msg), "%d %d", MSG_SET_WORDS_DIFFICULTY, context->lobby->words_difficulty);
468 send_tcp(context->sock, msg);
469 printf("Switched difficulty to: %d\n", context->lobby->words_difficulty);
470 } else if (button == btn_nb_assassin_switch) {
471 /* Alterne 1 -> 2 -> 3 -> 1 */
472 int nb = context->lobby->nb_assassins;
473 if (nb < 1 || nb > 3) nb = 1;

475 context->lobby->nb_assassins = nb;

481 }
482 return BTN_NONE;

740int struct_lobby_init(Lobby* lobby, int id, const char* code) {
