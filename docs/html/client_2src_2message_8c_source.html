9Arguments parse_arguments(char* message) {
10 Arguments args;
11 char* token = strtok(message, " ");
12 args.argc = 0;
13 args.argv = NULL;
14
15 while (token != NULL) {
16 args.argc++;
17 args.argv = realloc(args.argv, args.argc * sizeof(char*));
18 ((char**)args.argv)[args.argc - 1] = token;
19 token = strtok(NULL, " ");
20 }
21 return args;
22}
