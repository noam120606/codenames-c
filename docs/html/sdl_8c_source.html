205SDL_Texture* load_image(SDL_Renderer* renderer, const char* path) {
206 SDL_Surface* image = IMG_Load(path);
207 if (!image) {
208 printf("Erreur de chargement de l'image : %s\n", IMG_GetError());
209 return NULL;
210 }
211 SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer, image);
212 if (!texture) {
213 printf("Erreur de création de la texture : %s\n", SDL_GetError());
214 SDL_FreeSurface(image);
215 return NULL;
216 }
217 int tex_w = 0, tex_h = 0;
218 SDL_QueryTexture(texture, NULL, NULL, &tex_w, &tex_h);
219 if (tex_w == 0 || tex_h == 0) {
220 printf("Image '%s' failed to load : %dx%d\n", path, tex_w, tex_h);
221 }

226int display_image(SDL_Renderer* renderer, SDL_Texture* texture, int x, int y, float size_factor, double angle, SDL_RendererFlip flip, float ratio, Uint8 opacity) {
227 if (!renderer || !texture) {
228 return EXIT_FAILURE;
229 }
230
231 int og_w = 0, og_h = 0;
232 if (SDL_QueryTexture(texture, NULL, NULL, &og_w, &og_h) != 0) {
233 printf("SDL_QueryTexture Error: %s\n", SDL_GetError());
234 return EXIT_FAILURE;
235 }
236
237 SDL_Rect dstrect;
238 dstrect.x = (WIN_WIDTH - og_w*size_factor)/2 + x;
239 dstrect.y = (WIN_HEIGHT - og_h*size_factor)/2 - y; /* Inversion de l'axe y (seulement)*/
240
241 int final_w = og_w;
242 int final_h = og_h;
243
244 /* Calcul des ratios si altérés */
245 if (ratio != 1) {
246 final_w = og_w * ratio;
247 dstrect.x = (WIN_WIDTH - final_w*size_factor)/2 + x;
248 }
249 dstrect.w = (ratio != 1) ? final_w*size_factor : og_w*size_factor;
250 dstrect.h = (ratio != 1) ? final_h*size_factor : og_h*size_factor;
251
252 /* Point de pivot au centre de l'image */
253 SDL_Point center = {dstrect.w / 2, dstrect.h / 2};
254
255 /* Appliquer l'opacité */
256 SDL_SetTextureAlphaMod(texture, opacity);
257
258 /* Utiliser SDL_RenderCopyEx pour supporter la rotation et le flip */
259 SDL_RenderCopyEx(renderer, texture, NULL, &dstrect, angle, &center, flip);
260
261 return EXIT_SUCCESS;
262}

270void toggle_fullscreen(AppContext* context) {
271 if (!context || !context->window) return;
272
273 Uint32 now = SDL_GetTicks();
274 if (context->last_fullscreen_toggle_ms != 0 && now - context->last_fullscreen_toggle_ms < 180) {
275 return;
276 }
277 context->last_fullscreen_toggle_ms = now;
278
279 Uint32 flags = SDL_GetWindowFlags(context->window);
280 int is_fullscreen = (flags & SDL_WINDOW_FULLSCREEN_DESKTOP) || (flags & SDL_WINDOW_FULLSCREEN);
281
282 if (!is_fullscreen) {
283 int is_maximized = (flags & SDL_WINDOW_MAXIMIZED) != 0;
284 if (!is_maximized) {
285 int w = 0, h = 0;
286 int x = 0, y = 0;
287 SDL_GetWindowSize(context->window, &w, &h);
288 SDL_GetWindowPosition(context->window, &x, &y);
289 if (w >= WIN_MIN_WIDTH && h >= WIN_MIN_HEIGHT) {
290 context->windowed_width = w;
291 context->windowed_height = h;
292 }
293 context->windowed_x = x;
294 context->windowed_y = y;
295 }
296 }
297
298 SDL_SetWindowFullscreen(context->window, is_fullscreen ? 0 : SDL_WINDOW_FULLSCREEN_DESKTOP);
299
300 if (is_fullscreen) {
301 /* Sous Windows, on normalise explicitement l'état fenêtré pour éviter
302 les incohérences de bouton maximiser/restaurer après un aller-retour. */
303 #ifdef _WIN32
304 SDL_RestoreWindow(context->window);
305 #endif
306 if (context->windowed_width >= WIN_MIN_WIDTH && context->windowed_height >= WIN_MIN_HEIGHT) {
307 SDL_SetWindowSize(context->window, context->windowed_width, context->windowed_height);
308 }
309 if (context->windowed_x >= 0 && context->windowed_y >= 0) {
310 SDL_SetWindowPosition(context->window, context->windowed_x, context->windowed_y);
311 } else {
312 SDL_SetWindowPosition(context->window, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED);
313 }
314 }
