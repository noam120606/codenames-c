169SDL_Texture* load_image(SDL_Renderer* renderer, const char* path) {
170 SDL_Surface* image = IMG_Load(path);
171 if (!image) {
172 printf("Erreur de chargement de l'image : %s\n", IMG_GetError());
173 return NULL;
174 }
175 SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer, image);
176 if (!texture) {
177 printf("Erreur de création de la texture : %s\n", SDL_GetError());
178 SDL_FreeSurface(image);
179 return NULL;
180 }
181 int tex_w = 0, tex_h = 0;
182 SDL_QueryTexture(texture, NULL, NULL, &tex_w, &tex_h);
183 if (tex_w == 0 || tex_h == 0) {
184 printf("Image '%s' failed to load : %dx%d\n", path, tex_w, tex_h);
185 }
186 SDL_FreeSurface(image);
187 return texture;
188}

190int display_image(SDL_Renderer* renderer, SDL_Texture* texture, int x, int y, float size_factor, double angle, SDL_RendererFlip flip, float ratio, Uint8 opacity) {
191 if (!renderer || !texture) {
192 return EXIT_FAILURE;
193 }
194
195 int og_w = 0, og_h = 0;
196 if (SDL_QueryTexture(texture, NULL, NULL, &og_w, &og_h) != 0) {
197 printf("SDL_QueryTexture Error: %s\n", SDL_GetError());
198 return EXIT_FAILURE;
199 }
200
201 SDL_Rect dstrect;
202 dstrect.x = (WIN_WIDTH - og_w*size_factor)/2 + x;
203 dstrect.y = (WIN_HEIGHT - og_h*size_factor)/2 - y; /* Inversion de l'axe y (seulement)*/
204
205 int final_w = og_w;
206 int final_h = og_h;
207
208 /* Calcul des ratios si altérés */
209 if (ratio != 1) {
210 final_w = og_w * ratio;
211 dstrect.x = (WIN_WIDTH - final_w*size_factor)/2 + x;
212 }
213 dstrect.w = (ratio != 1) ? final_w*size_factor : og_w*size_factor;
214 dstrect.h = (ratio != 1) ? final_h*size_factor : og_h*size_factor;
215
216 /* Point de pivot au centre de l'image */
217 SDL_Point center = {dstrect.w / 2, dstrect.h / 2};
218
219 /* Appliquer l'opacité */
220 SDL_SetTextureAlphaMod(texture, opacity);
221
222 /* Utiliser SDL_RenderCopyEx pour supporter la rotation et le flip */
223 SDL_RenderCopyEx(renderer, texture, NULL, &dstrect, angle, &center, flip);
224
225 return EXIT_SUCCESS;
226}
