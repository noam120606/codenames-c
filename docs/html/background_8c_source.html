150void display_background(AppContext* context) {
151 const int INTERACT_RADIUS = 150;
152 const float MAX_SCALE = 1.4f;
153
154 const int cols = (WIN_WIDTH / BG_TILE_W) + BG_EXTRA;
155 const int rows = (WIN_HEIGHT / BG_TILE_H) + BG_EXTRA;
156
157 const int scaled_time = (int)(context->clock * BG_SPEED);
158 const int offset_x = scaled_time % (BG_TILE_W * 2);
159 const int offset_y = scaled_time % (BG_TILE_H * 2);
160 const int phase_x = world_phase(scaled_time, BG_TILE_W);
161 const int phase_y = world_phase(scaled_time, BG_TILE_H);
162
163 int raw_mx, raw_my;
164 SDL_GetMouseState(&raw_mx, &raw_my);
165 int logical_mx = raw_mx;
166 int logical_my = raw_my;
167 window_to_logical(context, raw_mx, raw_my, &logical_mx, &logical_my);
168 const int mouse_x = logical_mx - WIN_WIDTH / 2;
169 const int mouse_y = (WIN_HEIGHT / 2) - logical_my;
170
171 for (int i = -BG_EXTRA; i < cols; i++) {
172 const int row_offset = (i % 2) ? BG_TILE_H / 2 : 0;
173 const int wi = i + phase_x;
174
175 for (int j = -BG_EXTRA; j < rows; j++) {
176 const int x = (i * BG_TILE_W) - offset_x;
177 const int y = (j * BG_TILE_H) - offset_y + row_offset;
178 const int wj = j + phase_y;
179
180 /* Symbole : override éventuel, sinon distribution par défaut */
181 int idx = find_override(wi, wj);
182 int sym = (idx >= 0) ? overrides[idx].symbol : default_symbol(wi, wj);
183 SDL_Texture* symbol = symbol_texture(sym);
184 if (!symbol) continue;
185
186 /* Effet grossissement souris */
187 const int dx = x + (int)(BG_SYMBOL_W * BG_SIZE / 2) - mouse_x;
188 const int dy = y + (int)(BG_SYMBOL_H * BG_SIZE / 2) - mouse_y;
189 const int dist = (int)sqrt((double)(dx*dx + dy*dy));
190 float scale = 1.0f;
191 if (dist < INTERACT_RADIUS)
192 scale = 1.0f + (1.0f - (float)dist / INTERACT_RADIUS) * (MAX_SCALE - 1.0f);
193
194 int anim_idx = find_animation(wi, wj);
195 if (anim_idx >= 0) {
196 long elapsed = context->clock - animations[anim_idx].start_clock;
197 if (elapsed >= BG_ANIM_FRAMES) {
198 animations[anim_idx].active = 0;
199 } else {
200 float t = (float)elapsed / (float)BG_ANIM_FRAMES;
201 float pulse = 1.0f + ((t <= 0.5f ? t : (1.0f - t)) * 2.0f) * BG_ANIM_POP;
202 Uint8 old_opacity = (Uint8)(255.0f * (1.0f - t));
203 Uint8 new_opacity = (Uint8)(255.0f * t);
204 SDL_Texture* from_texture = symbol_texture(animations[anim_idx].from_symbol);
205 SDL_Texture* to_texture = symbol_texture(animations[anim_idx].to_symbol);
206
207 if (from_texture) {
208 display_image(context->renderer, from_texture, x, y, BG_SIZE * scale * pulse, 0,
209 row_offset == 0 ? SDL_FLIP_NONE : SDL_FLIP_HORIZONTAL, 1, old_opacity);
210 }
211 if (to_texture) {
212 display_image(context->renderer, to_texture, x, y, BG_SIZE * scale * pulse, 0,
213 row_offset == 0 ? SDL_FLIP_NONE : SDL_FLIP_HORIZONTAL, 1, new_opacity);
214 }
215 continue;
216 }
217 }
218
219 display_image(context->renderer, symbol, x, y, BG_SIZE * scale, 0,
220 row_offset == 0 ? SDL_FLIP_NONE : SDL_FLIP_HORIZONTAL, 1, 255);
221 }
222 }

226void background_handle_event(AppContext* context, SDL_Event* e) {
227 if (e->type != SDL_MOUSEBUTTONDOWN || e->button.button != SDL_BUTTON_LEFT) return;
228
229 const int scaled_time = (int)(context->clock * BG_SPEED);
230 const int offset_x = scaled_time % (BG_TILE_W * 2);
231 const int offset_y = scaled_time % (BG_TILE_H * 2);
232 const int phase_x = world_phase(scaled_time, BG_TILE_W);
233 const int phase_y = world_phase(scaled_time, BG_TILE_H);
234
235 /* Avec SDL_RenderSetLogicalSize, les coordonnées d'événement souris sont déjà logiques.
236 Les reconvertir ici introduit un décalage en plein écran (double scaling). */
237 const int logical_click_x = e->button.x;
238 const int logical_click_y = e->button.y;
239
240 /* Coordonnées clic dans le repère centré de display_image */
241 const int click_x = logical_click_x - WIN_WIDTH / 2;
242 const int click_y = (WIN_HEIGHT / 2) - logical_click_y;
243
244 const int cols = (WIN_WIDTH / BG_TILE_W) + BG_EXTRA;
245 const int rows = (WIN_HEIGHT / BG_TILE_H) + BG_EXTRA;
246
247 int best_dist2 = (BG_TILE_W / 2) * (BG_TILE_W / 2);
248 int best_wi = 0, best_wj = 0;
249 int found = 0;
250
251 for (int i = -BG_EXTRA; i < cols; i++) {
252 const int row_offset = (i % 2) ? BG_TILE_H / 2 : 0;
253 const int wi = i + phase_x;
254
255 for (int j = -BG_EXTRA; j < rows; j++) {
256 const int x = (i * BG_TILE_W) - offset_x;
257 const int y = (j * BG_TILE_H) - offset_y + row_offset;
258 const int cx = x + (int)(BG_SYMBOL_W * BG_SIZE / 2);
259 const int cy = y + (int)(BG_SYMBOL_H * BG_SIZE / 2);
260 const int dx = cx - click_x;
261 const int dy = cy - click_y;
262 const int dist2 = dx*dx + dy*dy;
263 if (dist2 < best_dist2) {
264 best_dist2 = dist2;
265 best_wi = wi;
266 best_wj = j + phase_y;
267 found = 1;
268 }
269 }
270 }
271
272 if (!found) return;
273
274 /* Cycle : lens(0) hat(1) spyglasses(2) lens(0) */
275 int idx = find_override(best_wi, best_wj);
276 int current = (idx >= 0) ? overrides[idx].symbol : default_symbol(best_wi, best_wj);
277 int next = (current + 1) % 3;
278
279 set_override(best_wi, best_wj, next);
280 start_animation(best_wi, best_wj, current, next, context->clock);
281}
