112static void card_start_flip_animation(AppContext* context, Card* card) {
113 int index = card_get_index(context, card);
114 if (index < 0) return;
115
116 CardFlipState* flip_state = &card_flip_states[index];
117 Uint32 now = SDL_GetTicks();
118
119 if (flip_state->animating) {
120 Uint32 elapsed = now - flip_state->started_at_ms;
121 float progress = (CARD_FLIP_DURATION_MS > 0)
122 ? ((float)elapsed / (float)CARD_FLIP_DURATION_MS)
123 : 1.0f;
124 if (progress > 1.0f) progress = 1.0f;
125
126 Booleen visible_side_is_word = (progress < 0.5f)
127 ? flip_state->from_show_word
128 : flip_state->to_show_word;
129
130 flip_state->from_show_word = visible_side_is_word;
131 flip_state->to_show_word = !visible_side_is_word;
132 flip_state->started_at_ms = now;
133 return;
134 }
135
136 flip_state->animating = True;
137 flip_state->from_show_word = card->display_word_once_revealed;
138 flip_state->to_show_word = !card->display_word_once_revealed;
139 flip_state->started_at_ms = now;
140}

142static int card_handle_click(AppContext* context, Card* card, Booleen is_guess_icon) {
143 if (!context || !card) return EXIT_FAILURE;
144
145 if (card->selected && is_guess_icon && !card->revealed) {
146 printf("Card \"%s\" guessed!\n", card->word);
147
148 char msg[64];
149 const char* agent_name = (context->player_name && context->player_name[0] != '\0') ? context->player_name : "Unknown";
150 format_to(msg, sizeof(msg), "%d %d %s", MSG_GUESS_CARD, (int)(card - context->lobby->game->cards), agent_name);
151 if (send_tcp(context->sock, msg) != EXIT_SUCCESS) {
152 printf("Failed to send click_card message to server\n");
153 return EXIT_FAILURE;
154 }
155
156 card->selected = False;
157 card->revealed = True;
158 } else if (!card->selected && !is_guess_icon && !card->revealed) {
159 card->selected = !card->selected;
160 char msg[64];
161 format_to(msg, sizeof(msg), "%d %d %d", MSG_PREGUESS, (int)(card - context->lobby->game->cards), card->selected);
162 send_tcp(context->sock, msg);
163 } else if (card->revealed) {
164 card_start_flip_animation(context, card);
165 } else { // La carte est déjà sélectionnée mais pas devinée
166 card->display_word_once_revealed = !card->display_word_once_revealed;
167 }
168
169 return EXIT_SUCCESS;
170}

180static int card_handle_event(AppContext* context, SDL_Event* event, Card* card) {
181 if (!context || !event || !card) return EXIT_FAILURE;
182
183 if (event->type == SDL_MOUSEMOTION) {
184 int mouseX = event->motion.x;
185 int mouseY = event->motion.y;
186 card->is_hovered = is_mouse_over_card(card, mouseX, mouseY);
187 } else if (event->type == SDL_MOUSEBUTTONDOWN && event->button.button == SDL_BUTTON_LEFT) {
188 int mouseX = event->button.x;
189 int mouseY = event->button.y;
190 int is_over = is_mouse_over_card(card, mouseX, mouseY);
191 card->is_hovered = is_over;
192 card->is_pressed = is_over;
193 } else if (event->type == SDL_MOUSEBUTTONUP && event->button.button == SDL_BUTTON_LEFT) {
194 int mouseX = event->button.x;
195 int mouseY = event->button.y;
196 int is_over = is_mouse_over_card(card, mouseX, mouseY);
197 int should_trigger = card->is_pressed && is_over;
198
199 card->is_hovered = is_over;
200 card->is_pressed = False;
201
202 if (should_trigger) {
203 audio_play(SOUND_BUTTON_CLICKED, 0);
204 return card_handle_click(context, card, is_mouse_over_guess_icon(card, mouseX, mouseY));
205 }
206 }
207 return EXIT_SUCCESS;
208}

293static int card_render(AppContext* context, Card* card, int x, int y, int index) {
294 if (!context || !card) return EXIT_FAILURE;
295
296 const int CARD_WIDTH = 200;
297 const int CARD_HEIGHT = 128;
298
299 card->rect = (SDL_Rect){
300 .x = (WIN_WIDTH - CARD_WIDTH) / 2 + x,
301 .y = (WIN_HEIGHT - CARD_HEIGHT) / 2 - y,
302 .w = CARD_WIDTH,
303 .h = CARD_HEIGHT
304 };
305
306 if (card->is_pressed && card->is_hovered) {
307 card->rect.x += 2;
308 card->rect.y += 2;
309 card->rect.w -= 4;
310 card->rect.h -= 2;
311 } else if (card->is_hovered) {
312 card->rect.x -= 4;
313 card->rect.y -= 2;
314 card->rect.w += 8;
315 card->rect.h += 4;
316 }
317
318 SDL_Rect draw_rect = card->rect;
319 SDL_Texture* texture_to_render = get_card_texture(context, card);
320 Booleen render_word = !card->revealed;
321
322 if (card->revealed && index >= 0 && index < NUM_CARDS) {
323 CardFlipState* flip_state = &card_flip_states[index];
324
325 if (flip_state->animating) {
326 Uint32 elapsed = SDL_GetTicks() - flip_state->started_at_ms;
327 float progress = (CARD_FLIP_DURATION_MS > 0)
328 ? ((float)elapsed / (float)CARD_FLIP_DURATION_MS)
329 : 1.0f;
330
331 if (progress >= 1.0f) {
332 progress = 1.0f;
333 flip_state->animating = False;
334 card->display_word_once_revealed = flip_state->to_show_word;
335 }
336
337 Booleen show_word_side = (progress < 0.5f)
338 ? flip_state->from_show_word
339 : flip_state->to_show_word;
340
341 float width_scale = (progress < 0.5f)
342 ? (1.0f - (progress * 2.0f))
343 : ((progress - 0.5f) * 2.0f);
344
345 int draw_width = (int)((float)card->rect.w * width_scale);
346 if (draw_width < CARD_FLIP_MIN_WIDTH_PX) {
347 draw_width = CARD_FLIP_MIN_WIDTH_PX;
348 }
349
350 draw_rect.x = card->rect.x + ((card->rect.w - draw_width) / 2);
351 draw_rect.w = draw_width;
352
353 texture_to_render = show_word_side
354 ? get_spy_word_side_texture(card)
355 : get_revealed_card_texture(card);
356
357 render_word = show_word_side && width_scale > 0.70f;
358 } else if (card->display_word_once_revealed) {
359 texture_to_render = get_spy_word_side_texture(card);
360 render_word = True;
361 } else {
362 texture_to_render = get_revealed_card_texture(card);
363 render_word = False;
364 }
365 } else if (!card->revealed && index >= 0 && index < NUM_CARDS) {
366 card_flip_states[index].animating = False;
367 card->display_word_once_revealed = False;
368 }
369
370 if (texture_to_render) {
371 SDL_RenderCopyEx(context->renderer, texture_to_render, NULL, &draw_rect, 0, NULL, SDL_FLIP_NONE);
372 }
373
374 if (card->selected) {
375 const int GUESS_ICON_SIZE = 32;
376 card->guess_rect = (SDL_Rect){
377 .x = card->rect.x + card->rect.w - GUESS_ICON_SIZE,
378 .y = card->rect.y,
379 .w = GUESS_ICON_SIZE,
380 .h = GUESS_ICON_SIZE
381 };
382 SDL_RenderCopyEx(context->renderer, guess_icon, NULL, &card->guess_rect, 0, NULL, SDL_FLIP_NONE);
383 }
384
385 if (render_word) {
386 card_render_word(context, card, index, x, y);
387 }
388
389 return EXIT_SUCCESS;
390}
