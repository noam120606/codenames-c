378 } else {
379 audio_play(SOUND_BUTTON_CLICKED, 0);

382 }
383 }

387static void button_build_hover_text_texture(Button* button, SDL_Renderer* renderer) {
388 if (!button || !button->cfg || !renderer) return;
389 if (!button->cfg->hover_text) return;
390 if (!button->cfg->font_path) return;
391
392 /* Détruire l'ancienne texture si présente */
393 if (button->cfg->hover_text_texture) {
394 SDL_DestroyTexture(button->cfg->hover_text_texture);
395 button->cfg->hover_text_texture = NULL;
396 }
397
398 TTF_Font* font = TTF_OpenFont(button->cfg->font_path, 128);
399 if (!font) return;
400 SDL_Surface* surf = TTF_RenderUTF8_Blended(font, button->cfg->hover_text, button->cfg->color);
401 TTF_CloseFont(font);
402 if (!surf) return;
403
404 SDL_Texture* tex = SDL_CreateTextureFromSurface(renderer, surf);
405 int surf_w = surf->w;
406 int surf_h = surf->h;
407 SDL_FreeSurface(surf);
408 if (!tex) return;
409
410 /* Calculer la taille et la position du tooltip au-dessus du bouton */
411 int desired_h = (button->cfg->h * 40) / 100;
412 if (desired_h < 10) desired_h = 10;
413 double scale = (surf_h > 0) ? ((double)desired_h / (double)surf_h) : 1.0;
414 int draw_w = (int)(surf_w * scale);
415 int draw_h = desired_h;
416 int draw_x = button->cfg->x + (button->cfg->rect.w - draw_w) / 2;
417 int draw_y = button->cfg->y - draw_h - 6;
418
419 button->cfg->hover_text_texture = tex;
420 button->cfg->hover_text_rect = (SDL_Rect){draw_x, draw_y, draw_w, draw_h};
421}

432int button_render(SDL_Renderer* renderer, Button* button) {
433
434 if (!renderer || !button || !button->cfg) return EXIT_FAILURE;
435 ButtonConfig* cfg = button->cfg;
436 if (!cfg->texture) return EXIT_FAILURE;
437
438 if (cfg->renderer != renderer) cfg->renderer = renderer;
439 if (cfg->is_text && cfg->text_dirty) {
440 if (button_rebuild_text_texture(button, renderer) != EXIT_SUCCESS) {
441 return EXIT_FAILURE;
442 }
443 }
444
445 /* Grandir le bouton au survol */
446 SDL_Rect render_rect = cfg->rect;
447 SDL_Rect text_rect = cfg->text_rect;
448 if (cfg->is_pressed && cfg->is_hovered) {
449 /* Animation d'appui: léger enfoncement vers le bas et compression. */
450 render_rect.x += 2;
451 render_rect.y += 2;
452 render_rect.w -= 4;
453 render_rect.h -= 2;
454 if (cfg->is_text) {
455 text_rect.x += 2;
456 text_rect.y += 2;
457 text_rect.w -= 2;
458 text_rect.h -= 1;
459 }
460 } else if (cfg->is_hovered) {
461 render_rect.x -= 4;
462 render_rect.y -= 2;
463 render_rect.w += 8;
464 render_rect.h += 4;
465 if (cfg->is_text) {
466 text_rect.x -= 2;
467 text_rect.y -= 1;
468 text_rect.w += 4;
469 text_rect.h += 2;
470 }
471 }
472
473 SDL_RenderCopyEx(renderer, cfg->texture, NULL, &render_rect, 0, NULL, SDL_FLIP_NONE);
474 if (cfg->is_text) {
475 SDL_RenderCopyEx(renderer, cfg->text_texture, NULL, &text_rect, 0, NULL, SDL_FLIP_NONE);
476 }
477
478 /* Afficher le texte de hover (tooltip) après un délai configuré. */
479 if (cfg->is_hovered && cfg->hover_text) {
480 if (!cfg->hover_text_texture && cfg->renderer) {
481 Uint32 now = SDL_GetTicks();
482 if (cfg->hover_start_ticks == 0) {
483 cfg->hover_start_ticks = now;
484 }
485 if ((now - cfg->hover_start_ticks) >= cfg->hover_delay_ms) {
486 button_build_hover_text_texture(button, cfg->renderer);
487 }
488 }
489 if (cfg->hover_text_texture) {
490 SDL_RenderCopyEx(renderer, cfg->hover_text_texture, NULL, &cfg->hover_text_rect, 0, NULL, SDL_FLIP_NONE);
491 }
492 }
493
494 return EXIT_SUCCESS;
495}
