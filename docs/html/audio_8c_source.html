59/* Convertit une fréquence de coupure (Hz) en coefficient alpha du passe-bas. */
60static float compute_low_pass_alpha(float cutoff_hz) {
61 int frequency = 0;
62 Uint16 format = 0;
63 int channels = 0;
64
65 if (Mix_QuerySpec(&frequency, &format, &channels) == 0 || frequency <= 0) {
66 frequency = 44100;
67 }
68
69 if (cutoff_hz <= 0.0f) {
70 cutoff_hz = AUDIO_FILTER_DEFAULT_CUTOFF_HZ;
71 }
72
73 float dt = 1.0f / (float)frequency;
74 float rc = 1.0f / (2.0f * AUDIO_PI * cutoff_hz);
75 return dt / (rc + dt);
76}

88/* Callback SDL_mixer: applique un passe-bas fixe sur le flux stéréo du canal. */
89static void low_pass_effect(int channel, void* stream, int len, void* udata) {
90 (void)channel;
91
92 ChannelFilterState* state = (ChannelFilterState*)udata;
93 if (!state || !state->active || !stream || len <= 0) {
94 return;
95 }
96
97 Sint16* samples = (Sint16*)stream;
98 int sample_count = len / (int)sizeof(Sint16);
99
100 for (int i = 0; i < sample_count; i += 2) {
101 float left_in = (float)samples[i];
102 state->previous_left += state->alpha * (left_in - state->previous_left);
103 samples[i] = (Sint16)state->previous_left;
104
105 if (i + 1 < sample_count) {
106 float right_in = (float)samples[i + 1];
107 state->previous_right += state->alpha * (right_in - state->previous_right);
108 samples[i + 1] = (Sint16)state->previous_right;
109 }
110 }
111}

113/* Callback SDL_mixer: applique un passe-bas interpolé pour un fade par filtre. */
114static void fade_low_pass_effect(int channel, void* stream, int len, void* udata) {
115 (void)channel;
116
117 ChannelFadeFilterState* state = (ChannelFadeFilterState*)udata;
118 if (!state || !state->active || !stream || len <= 0) {
119 return;
120 }
121
122 Uint32 now = SDL_GetTicks();
123 Uint32 elapsed = now - state->start_ticks;
124 float t = 1.0f;
125 if (state->duration_ms > 0) {
126 t = (float)elapsed / (float)state->duration_ms;
127 if (t < 0.0f) t = 0.0f;
128 if (t > 1.0f) t = 1.0f;
129 }
130
131 float alpha = state->start_alpha + (state->end_alpha - state->start_alpha) * t;
132
133 Sint16* samples = (Sint16*)stream;
134 int sample_count = len / (int)sizeof(Sint16);
135
136 for (int i = 0; i < sample_count; i += 2) {
137 float left_in = (float)samples[i];
138 state->previous_left += alpha * (left_in - state->previous_left);
139 samples[i] = (Sint16)state->previous_left;
140
141 if (i + 1 < sample_count) {
142 float right_in = (float)samples[i + 1];
143 state->previous_right += alpha * (right_in - state->previous_right);
144 samples[i + 1] = (Sint16)state->previous_right;
145 }
146 }
147
148 if (t >= 1.0f && state->deactivate_on_complete) {
149 state->active = 0;
150 }
151}

175/* (Ré)applique le filtre configuré pour un son sur son canal de lecture. */
176static void apply_channel_filter(int channel, SoundID id) {
177 if (channel < 0 || channel >= MAX_SOUNDS || !is_valid_sound_id(id)) {
178 return;
179 }
180
181 float new_alpha = compute_low_pass_alpha(sound_cfgs[id].filter_cutoff_hz);
182 int want_filter = !sound_cfgs[id].bypass_filter;
183
184 /* Si le filtre est déjà actif avec le même alpha, ne rien faire
185 pour ne pas réinitialiser l'état du filtre (previous_left/right). */
186 if (channel_filters[channel].active == want_filter) {
187 if (!want_filter) {
188 return; /* Pas de filtre demandé, pas de filtre actif → OK */
189 }
190 if (channel_filters[channel].alpha == new_alpha) {
191 return; /* Même filtre déjà appliqué → OK */
192 }
193 }
194
195 Mix_UnregisterAllEffects(channel);
196 channel_filters[channel].active = 0;
197 channel_filters[channel].previous_left = 0.0f;
198 channel_filters[channel].previous_right = 0.0f;
199
200 if (!want_filter) {
201 return;
202 }
203
204 channel_filters[channel].alpha = new_alpha;
205 channel_filters[channel].active = 1;
206
207 if (Mix_RegisterEffect(channel, low_pass_effect, NULL, &channel_filters[channel]) == 0) {
208 printf("Erreur Mix_RegisterEffect: %s\n", Mix_GetError());
209 channel_filters[channel].active = 0;
210 }

377 channel_fade_filters[channel].previous_left = 0.0f;

381 printf("Erreur Mix_RegisterEffect: %s\n", Mix_GetError());
382 Mix_HaltChannel(channel);

384 sound_channels[id] = -1;
385 return EXIT_FAILURE;
386 }

394 return EXIT_FAILURE;
