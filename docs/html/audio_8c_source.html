40static float compute_low_pass_alpha(float cutoff_hz) {
41 int frequency = 0;
42 Uint16 format = 0;
43 int channels = 0;
44
45 if (Mix_QuerySpec(&frequency, &format, &channels) == 0 || frequency <= 0) {
46 frequency = 44100;
47 }
48
49 if (cutoff_hz <= 0.0f) {
50 cutoff_hz = FILTER_DEFAULT_CUTOFF_HZ;
51 }
52
53 float dt = 1.0f / (float)frequency;
54 float rc = 1.0f / (2.0f * AUDIO_PI * cutoff_hz);
55 return dt / (rc + dt);
56}

71static void low_pass_effect(int channel, void* stream, int len, void* udata) {
72 (void)channel;
73
74 ChannelFilterState* state = (ChannelFilterState*)udata;
75 if (!state || !state->active || !stream || len <= 0) {
76 return;
77 }
78
79 Sint16* samples = (Sint16*)stream;
80 int sample_count = len / (int)sizeof(Sint16);
81
82 for (int i = 0; i < sample_count; i += 2) {
83 float left_in = (float)samples[i];
84 state->previous_left += state->alpha * (left_in - state->previous_left);
85 samples[i] = (Sint16)state->previous_left;
86
87 if (i + 1 < sample_count) {
88 float right_in = (float)samples[i + 1];
89 state->previous_right += state->alpha * (right_in - state->previous_right);
90 samples[i + 1] = (Sint16)state->previous_right;
91 }
92 }
93}

107static void apply_channel_filter(int channel, SoundID id) {
108 if (channel < 0 || channel >= MAX_SOUNDS || !is_valid_sound_id(id)) {
109 return;
110 }
111
112 float new_alpha = compute_low_pass_alpha(sound_cfgs[id].filter_cutoff_hz);
113 int want_filter = !sound_cfgs[id].bypass_filter;
114
115 /* Si le filtre est déjà actif avec le même alpha, ne rien faire
116 pour ne pas réinitialiser l'état du filtre (previous_left/right). */
117 if (channel_filters[channel].active == want_filter) {
118 if (!want_filter) {
119 return; /* Pas de filtre demandé, pas de filtre actif → OK */
120 }
121 if (channel_filters[channel].alpha == new_alpha) {
122 return; /* Même filtre déjà appliqué → OK */
123 }
124 }
125
126 Mix_UnregisterAllEffects(channel);
127 channel_filters[channel].active = 0;
128 channel_filters[channel].previous_left = 0.0f;
129 channel_filters[channel].previous_right = 0.0f;
130
131 if (!want_filter) {
132 return;
133 }
134
135 channel_filters[channel].alpha = new_alpha;
136 channel_filters[channel].active = 1;
137
138 if (Mix_RegisterEffect(channel, low_pass_effect, NULL, &channel_filters[channel]) == 0) {
139 printf("Erreur Mix_RegisterEffect: %s\n", Mix_GetError());
140 channel_filters[channel].active = 0;
141 }
142}
