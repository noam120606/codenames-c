56static int input_utf8_width_range(TTF_Font* font, char* text, int start, int end) {
57 if (!font || !text) return 0;
58
59 int len = (int)strlen(text);
60 if (start < 0) start = 0;
61 if (end < start) end = start;
62 if (start > len) start = len;
63 if (end > len) end = len;
64 if (end == start) return 0;
65
66 char saved = text[end];
67 text[end] = '\0';
68
69 int w = 0;
70 TTF_SizeUTF8(font, text + start, &w, NULL);
71
72 text[end] = saved;
73 return w;

76static void input_ensure_cursor_visible(Input* in, TTF_Font* font, int content_width) {
77 if (!in || !in->cfg || !in->cfg->text || !font) return;
78
79 if (content_width <= 0) {
80 in->cfg->view_start = in->cfg->cursor_pos;
81 return;
82 }
83
84 int len = in->cfg->len;
85 int cursor = in->cfg->cursor_pos;
86 if (cursor < 0) cursor = 0;
87 if (cursor > len) cursor = len;
88 in->cfg->cursor_pos = cursor;
89
90 int start = in->cfg->view_start;
91 if (start < 0) start = 0;
92 if (start > len) start = len;
93 if (start > cursor) start = cursor;
94
95 int cursor_w = input_utf8_width_range(font, in->cfg->text, start, cursor);
96 while (cursor_w > content_width && start < cursor) {
97 start = utf8_next_char_pos(in->cfg->text, len, start);
98 cursor_w = input_utf8_width_range(font, in->cfg->text, start, cursor);
99 }
100
101 while (start > 0) {
102 int prev = utf8_prev_char_pos(in->cfg->text, start);
103 int w = input_utf8_width_range(font, in->cfg->text, prev, cursor);
104 if (w <= content_width) start = prev;
105 else break;
106 }
107
108 in->cfg->view_start = start;
109}

318static void input_history_push(Input* in, const char* submitted_text) {
319 if (!in || !in->cfg) return;
320
321 if (!submitted_text || submitted_text[0] == '\0') {
322 input_reset_history_navigation(in);
323 return;
324 }
325
326 if (in->cfg->history_count > 0) {
327 const char* last = in->cfg->history[in->cfg->history_count - 1];
328 if (last && strcmp(last, submitted_text) == 0) {
329 input_reset_history_navigation(in);
330 return;
331 }
332 }
333
334 char* copy = strdup(submitted_text);
335 if (!copy) {
336 input_reset_history_navigation(in);
337 return;
338 }
339
340 if (in->cfg->history_count >= INPUT_HISTORY_MAX) {
341 free(in->cfg->history[0]);
342 memmove(&in->cfg->history[0], &in->cfg->history[1], sizeof(in->cfg->history[0]) * (INPUT_HISTORY_MAX - 1));
343 in->cfg->history[INPUT_HISTORY_MAX - 1] = copy;
344 in->cfg->history_count = INPUT_HISTORY_MAX;
345 } else {
346 in->cfg->history[in->cfg->history_count] = copy;
347 in->cfg->history_count++;
348 }
349
350 input_reset_history_navigation(in);
351}

353static int input_history_navigate(Input* in, int direction) {
354 if (!in || !in->cfg || in->cfg->history_count <= 0) return 0;
355
356 if (direction < 0) {
357 if (in->cfg->history_index == in->cfg->history_count) {
358 if (in->cfg->history_draft_text) {
359 free(in->cfg->history_draft_text);
360 in->cfg->history_draft_text = NULL;
361 }
362 in->cfg->history_draft_text = strdup(in->cfg->text ? in->cfg->text : "");
363 }
364
365 if (in->cfg->history_index > 0) in->cfg->history_index--;
366 } else if (direction > 0) {
367 if (in->cfg->history_index >= in->cfg->history_count) return 0;
368 in->cfg->history_index++;
369 } else {
370 return 0;
371 }
372
373 if (in->cfg->history_index >= in->cfg->history_count) {
374 const char* draft = in->cfg->history_draft_text ? in->cfg->history_draft_text : "";
375 input_set_text_internal(in, draft, 0);
376 if (in->cfg->history_draft_text) {
377 free(in->cfg->history_draft_text);
378 in->cfg->history_draft_text = NULL;
379 }
380 } else {
381 const char* history_value = in->cfg->history[in->cfg->history_index];
382 input_set_text_internal(in, history_value ? history_value : "", 0);
383 }
384
385 return 1;
386}

420Input* input_create(SDL_Renderer* renderer, InputId id, const InputConfig* cfg_in) {
421 if (input_count >= MAX_INPUTS) return NULL;
422
423 Input* in = (Input*)malloc(sizeof(Input));
424 if (!in) return NULL;
425 in->id = id;
426
427 /* Input owns its config. We copy cfg_in (or defaults) into a freshly allocated struct,
428 * then allocate runtime buffers (text) and reset pointers (textures). */
429 in->cfg = input_config_init();
430 if (!in->cfg) {
431 free(in);
432 return NULL;
433 }
434 if (cfg_in) {
435 /* Shallow copy first, then fix ownership-sensitive fields below. */
436 *in->cfg = *cfg_in;
437 }
438
439 /* runtime-managed fields: always start from a clean state */
440 in->cfg->bg_texture = NULL;
441 in->cfg->submitted_text = NULL;
442 in->cfg->text = NULL;
443 in->cfg->len = 0;
444 in->cfg->cursor_pos = 0;
445 in->cfg->view_start = 0;
446 in->cfg->focused = 0;
447 in->cfg->submitted = 0;
448 in->cfg->sel_start = 0;
449 in->cfg->sel_len = 0;
450 in->cfg->sel_anchor = 0;
451 for (int i = 0; i < INPUT_HISTORY_MAX; i++) in->cfg->history[i] = NULL;
452 in->cfg->history_count = 0;
453 in->cfg->history_index = 0;
454 in->cfg->history_draft_text = NULL;
455 in->cfg->placeholder_index = 0;
456 in->cfg->placeholder_last_tick = SDL_GetTicks();
457 in->cfg->submit_sound_chunk = NULL;
458
459 /* Conversion coordonnées relatives au centre -> écran.
460 * x positif = droite, y positif = haut.
461 * (x, y) représente le centre de l'input relatif au centre de la fenêtre. */
462 int rel_x = in->cfg->x;
463 int rel_y = in->cfg->y;
464 int screen_x = (WIN_WIDTH / 2) + rel_x - (in->cfg->w / 2);
465 int screen_y = (WIN_HEIGHT / 2) - rel_y - (in->cfg->h / 2);
466
467 in->cfg->x = screen_x;
468 in->cfg->y = screen_y;
469
470 /* keep rect in sync */
471 in->cfg->rect = (SDL_Rect){in->cfg->x, in->cfg->y, in->cfg->w, in->cfg->h};
472
473 /* allocate text buffer */
474 if (in->cfg->maxlen <= 0) in->cfg->maxlen = INPUT_DEFAULT_MAX;
475 in->cfg->text = (char*)calloc((size_t)in->cfg->maxlen + 1, 1);
476 if (!in->cfg->text) {
477 input_destroy(in);
478 return NULL;
479 }
480
481 /* placeholders: just reference caller-provided strings (do not strdup),
482 * so stack/const arrays are safe as long as they outlive the input.
483 * (This matches current usage in menu.c.) */
484 /* submitted_label is a const char*; no allocation here */
485
486 if (in->cfg->init_text && in->cfg->init_text[0] != '\0') {
487 strncpy(in->cfg->text, in->cfg->init_text, (size_t)in->cfg->maxlen);
488 in->cfg->text[in->cfg->maxlen] = '\0';
489 in->cfg->len = (int)strlen(in->cfg->text);
490 in->cfg->cursor_pos = in->cfg->len;
491 in->cfg->view_start = 0;
492 }
493
494 /* Load saved player name from properties file if persistence is enabled */
495 if (in->cfg->save_player_data && in->id == INPUT_NAME) {
496 char saved_name[INPUT_DEFAULT_MAX + 1] = {0};
497 if (read_property(saved_name, "PLAYER_NAME") == EXIT_SUCCESS && saved_name[0] != '\0') {
498 strncpy(in->cfg->text, saved_name, (size_t)in->cfg->maxlen);
499 in->cfg->text[in->cfg->maxlen] = '\0';
500 in->cfg->len = (int)strlen(in->cfg->text);
501 in->cfg->cursor_pos = in->cfg->len;
502 in->cfg->view_start = 0;
503 }
504 }
505
506 /* auto-load background image if bg_path was provided in config */
507 if (in->cfg->bg_path && renderer) {
508 input_set_bg(in, renderer, in->cfg->bg_path, in->cfg->bg_padding);
509 }
510
511 /* preload submit sound if configured */
512 input_reload_submit_sound(in);
513
514 inputs[input_count++] = in;
515
516 return in;
517}

522 for (int i = 0; i < input_count; ++i) {
523 if (inputs[i] == in) {
524 for (int j = i; j < input_count - 1; ++j) inputs[j] = inputs[j + 1];
525 inputs[input_count - 1] = NULL;
526 input_count--;
527 break;
528 }
529 }
530
531 if (in->cfg) {
532 if (in->cfg->text) free(in->cfg->text);
533 if (in->cfg->bg_texture) free_image(in->cfg->bg_texture);
534 if (in->cfg->submitted_text) free(in->cfg->submitted_text);

538 in->cfg->history[i] = NULL;
539 }
540 if (in->cfg->submit_sound_chunk) Mix_FreeChunk(in->cfg->submit_sound_chunk);
541 /* submitted_label and placeholders are not owned (const pointers) */
542 free(in->cfg);
543 }
544 free(in);
545}

549static int input_text_allowed(const char* allowed_pattern, const char* txt) {
550 if (!allowed_pattern || !txt) return 1;
551 int len = (int)strlen(txt);
552 for (int i = 0; i < len; i++) {
553 char ch[2] = { txt[i], '\0' };

565void input_handle_event(AppContext* context, Input* in, SDL_Event* e) {
566 if (!in || !e) return;
567
568 if (e->type == SDL_MOUSEBUTTONDOWN) {
569 int mx = e->button.x;
570 int my = e->button.y;
571 if (point_in_rect(mx, my, &in->cfg->rect)) {
572 in->cfg->focused = 1;
573 /* place cursor at end */
574 in->cfg->cursor_pos = in->cfg->len;
575 if (in->cfg->view_start > in->cfg->cursor_pos) in->cfg->view_start = in->cfg->cursor_pos;
576 input_sync_text_input_state();
577 } else {
578 if (in->cfg->focused) {
579 in->cfg->focused = 0;
580 input_sync_text_input_state();
581 }
582 }
583 }
584
585 if (!in->cfg->focused) return;
586
587 if (e->type == SDL_TEXTINPUT) {
588 const char* txt = e->text.text;
589 int add = strlen(txt);
590 if (in->cfg->len + add > in->cfg->maxlen) return;
591 /* filter characters against allowed_pattern if set */
592 if (!input_text_allowed(in->cfg->allowed_pattern, txt)) return;
593 /* if there is a selection, replace it */
594 if (input_has_selection(in)) {
595 input_delete_selection(in);
596 }
597 /* insert at cursor_pos */
598 memmove(in->cfg->text + in->cfg->cursor_pos + add, in->cfg->text + in->cfg->cursor_pos, in->cfg->len - in->cfg->cursor_pos + 1);
599 memcpy(in->cfg->text + in->cfg->cursor_pos, txt, add);
600 in->cfg->cursor_pos += add;
601 in->cfg->len += add;
602 input_reset_history_navigation(in);
603 }
604
605 if (e->type == SDL_KEYDOWN) {
606 SDL_Keycode k = e->key.keysym.sym;
607 SDL_Keymod mod = e->key.keysym.mod;
608 /* Ctrl+A = select all */
609 if ((mod & KMOD_CTRL) && (k == SDLK_a)) {
610 in->cfg->sel_start = 0;
611 in->cfg->sel_len = in->cfg->len;
612 in->cfg->sel_anchor = 0;
613 in->cfg->cursor_pos = in->cfg->len;
614 return;
615 }
616
617 if (k == SDLK_UP) {
618 if (input_history_navigate(in, -1)) return;
619 } else if (k == SDLK_DOWN) {
620 if (input_history_navigate(in, 1)) return;
621 }
622
623 if (k == SDLK_BACKSPACE) {
624 int text_changed = 0;
625 if (input_has_selection(in)) {
626 input_delete_selection(in);
627 text_changed = 1;
628 } else if (mod & KMOD_CTRL) {
629 int np = prev_word_pos(in, in->cfg->cursor_pos);
630 int del = in->cfg->cursor_pos - np;
631 if (del > 0) {
632 memmove(in->cfg->text + np, in->cfg->text + in->cfg->cursor_pos, in->cfg->len - in->cfg->cursor_pos + 1);
633 in->cfg->len -= del;
634 in->cfg->cursor_pos = np;
635 in->cfg->text[in->cfg->len] = '\0';
636 text_changed = 1;
637 }
638 } else {
639 if (in->cfg->cursor_pos > 0 && in->cfg->len > 0) {
640 int np = utf8_prev_char_pos(in->cfg->text, in->cfg->cursor_pos);
641 int del = in->cfg->cursor_pos - np;
642 memmove(in->cfg->text + np, in->cfg->text + in->cfg->cursor_pos, in->cfg->len - in->cfg->cursor_pos + 1);
643 in->cfg->cursor_pos = np;
644 in->cfg->len -= del;
645 in->cfg->text[in->cfg->len] = '\0';
646 text_changed = 1;
647 }
648 }
649 if (text_changed) input_reset_history_navigation(in);
650 input_clear_selection_internal(in);
651 } else if (k == SDLK_DELETE) {
652 int text_changed = 0;
653 if (input_has_selection(in)) {
654 input_delete_selection(in);
655 text_changed = 1;
656 } else if (mod & KMOD_CTRL) {
657 int np = next_word_pos(in, in->cfg->cursor_pos);
658 int del = np - in->cfg->cursor_pos;
659 if (del > 0) {
660 memmove(in->cfg->text + in->cfg->cursor_pos, in->cfg->text + np, in->cfg->len - np + 1);
661 in->cfg->len -= del;
662 in->cfg->text[in->cfg->len] = '\0';
663 text_changed = 1;
664 }
665 } else {
666 if (in->cfg->cursor_pos < in->cfg->len && in->cfg->len > 0) {
667 int np = utf8_next_char_pos(in->cfg->text, in->cfg->len, in->cfg->cursor_pos);
668 int del = np - in->cfg->cursor_pos;
669 memmove(in->cfg->text + in->cfg->cursor_pos, in->cfg->text + np, in->cfg->len - np + 1);
670 in->cfg->len -= del;
671 in->cfg->text[in->cfg->len] = '\0';
672 text_changed = 1;
673 }
674 }
675 if (text_changed) input_reset_history_navigation(in);
676 input_clear_selection_internal(in);
677 } else if (k == SDLK_LEFT) {
678 int newpos = in->cfg->cursor_pos;
679 if (mod & KMOD_CTRL) newpos = prev_word_pos(in, in->cfg->cursor_pos);
680 else if (in->cfg->cursor_pos > 0) newpos = utf8_prev_char_pos(in->cfg->text, in->cfg->cursor_pos);
681
682 if (mod & KMOD_SHIFT) {
683 if (!input_has_selection(in)) {
684 in->cfg->sel_anchor = in->cfg->cursor_pos;
685 in->cfg->sel_start = in->cfg->cursor_pos;
686 }
687 in->cfg->cursor_pos = newpos;
688 int s = in->cfg->sel_anchor;
689 int epos = in->cfg->cursor_pos;
690 if (epos < s) { in->cfg->sel_start = epos; in->cfg->sel_len = s - epos; }
691 else { in->cfg->sel_start = s; in->cfg->sel_len = epos - s; }
692 } else {
693 in->cfg->cursor_pos = newpos;
694 input_clear_selection_internal(in);
695 }
696 } else if (k == SDLK_RIGHT) {
697 int newpos = in->cfg->cursor_pos;
698 if (mod & KMOD_CTRL) newpos = next_word_pos(in, in->cfg->cursor_pos);
699 else if (in->cfg->cursor_pos < in->cfg->len) newpos = utf8_next_char_pos(in->cfg->text, in->cfg->len, in->cfg->cursor_pos);
700
701 if (mod & KMOD_SHIFT) {
702 if (!input_has_selection(in)) {
703 in->cfg->sel_anchor = in->cfg->cursor_pos;
704 in->cfg->sel_start = in->cfg->cursor_pos;
705 }
706 in->cfg->cursor_pos = newpos;
707 int s = in->cfg->sel_anchor;
708 int epos = in->cfg->cursor_pos;
709 if (epos < s) { in->cfg->sel_start = epos; in->cfg->sel_len = s - epos; }
710 else { in->cfg->sel_start = s; in->cfg->sel_len = epos - s; }
711 } else {
712 in->cfg->cursor_pos = newpos;
713 input_clear_selection_internal(in);
714 }
715 } else if (k == SDLK_RETURN || k == SDLK_KP_ENTER) {
716 input_submit_internal(context, in);
717 }
718 }

728void input_render(SDL_Renderer* renderer, Input* in) {
729 if (!renderer || !in) return;
730 SDL_Rect rect = (SDL_Rect){ in->cfg->x, in->cfg->y, in->cfg->w, in->cfg->h };
731 in->cfg->rect = rect;
732
733 /* background: image if present else color */
734 if (in->cfg->bg_texture) {
735 SDL_RenderCopy(renderer, in->cfg->bg_texture, NULL, &rect);
736 } else {
737 SDL_SetRenderDrawColor(renderer, in->cfg->bg_color.r, in->cfg->bg_color.g, in->cfg->bg_color.b, in->cfg->bg_color.a);
738 SDL_RenderFillRect(renderer, &rect);
739 }
740
741 /* border: draw only when no background texture is set */
742 if (!in->cfg->bg_texture) {
743 SDL_SetRenderDrawColor(renderer, in->cfg->border_color.r, in->cfg->border_color.g, in->cfg->border_color.b, in->cfg->border_color.a);
744 SDL_RenderDrawRect(renderer, &rect);
745 }
746
747 /* render submitted label (always if set) and submitted text (if present) above input */
748 if (in->cfg->font_path && in->cfg->submitted_label) {
749 int label_size = in->cfg->font_size > 12 ? in->cfg->font_size - 4 : in->cfg->font_size;
750 TTF_Font* label_font = TTF_OpenFont(in->cfg->font_path, label_size);
751 if (label_font) {
752 SDL_Color color = {255,255,255,255};
753 const char* lbl = in->cfg->submitted_label ? in->cfg->submitted_label : "";
754 const char* sub = in->cfg->submitted_text ? in->cfg->submitted_text : "";
755 size_t L = strlen(lbl);
756 size_t M = strlen(sub);
757 char* combined = (char*)malloc(L + M + 1);
758 if (combined) {
759 memcpy(combined, lbl, L);
760 memcpy(combined + L, sub, M + 1);
761 SDL_Surface* surf_label = TTF_RenderUTF8_Blended(label_font, combined, color);
762 if (surf_label) {
763 SDL_Texture* tex_label = SDL_CreateTextureFromSurface(renderer, surf_label);
764 if (tex_label) {
765 int tw = 0, th = 0;
766 SDL_QueryTexture(tex_label, NULL, NULL, &tw, &th);
767 int dst_x = in->cfg->x + in->cfg->padding;
768 int dst_y = in->cfg->y - th - 8;
769 SDL_Rect dst = { dst_x, dst_y, tw, th };
770 SDL_RenderCopy(renderer, tex_label, NULL, &dst);
771 SDL_DestroyTexture(tex_label);
772 }
773 SDL_FreeSurface(surf_label);
774 }
775 free(combined);
776 }
777 TTF_CloseFont(label_font);
778 }
779 }
780
781 /* render text using TTF */
782 if (in->cfg->font_path && in->cfg->font_size > 0) {
783 TTF_Font* font = TTF_OpenFont(in->cfg->font_path, in->cfg->font_size);
784 if (font) {
785 int padding = in->cfg->padding;
786 if (padding < 0) padding = 0;
787 int content_width = in->cfg->w - (padding * 2);
788 if (content_width < 0) content_width = 0;
789
790 int visible_start = 0;
791 int visible_end = 0;
792 int cursor_x_rel = 0;
793 int text_offset_x = padding;
794
795 if (in->cfg->len > 0 && in->cfg->text) {
796 input_ensure_cursor_visible(in, font, content_width);
797 visible_start = in->cfg->view_start;
798 visible_end = input_compute_visible_end(in, font, visible_start, content_width);
799 if (visible_end < visible_start) visible_end = visible_start;
800
801 int full_tw = input_utf8_width_range(font, in->cfg->text, 0, in->cfg->len);
802 int visible_tw = input_utf8_width_range(font, in->cfg->text, visible_start, visible_end);
803 if (in->cfg->centered && visible_start == 0 && visible_end == in->cfg->len && full_tw <= content_width) {
804 int center_off = (content_width - visible_tw) / 2;
805 if (center_off < 0) center_off = 0;
806 text_offset_x = padding + center_off;
807 }
808
809 cursor_x_rel = input_utf8_width_range(font, in->cfg->text, visible_start, in->cfg->cursor_pos);
810 } else {
811 in->cfg->view_start = 0;
812 visible_start = 0;
813 visible_end = 0;
814 if (in->cfg->centered && content_width > 0) {
815 text_offset_x = padding + (content_width / 2);
816 }
817 }
818
819 /* draw selection highlight if any */
820 if (input_has_selection(in)) {
821 int s = in->cfg->sel_start;
822 int l = in->cfg->sel_len;
823 if (s < 0) s = 0;
824 if (s > in->cfg->len) s = in->cfg->len;
825 if (l < 0) l = 0;
826 if (s + l > in->cfg->len) l = in->cfg->len - s;
827
828 int sel_start = s;
829 int sel_end = s + l;
830 if (sel_start < visible_start) sel_start = visible_start;
831 if (sel_end > visible_end) sel_end = visible_end;
832
833 if (sel_end > sel_start) {
834 int bx = input_utf8_width_range(font, in->cfg->text, visible_start, sel_start);
835 int sw = input_utf8_width_range(font, in->cfg->text, sel_start, sel_end);
836 int sh = TTF_FontHeight(font);
837 if (sh <= 0) sh = in->cfg->font_size;
838 SDL_Rect selrect = { in->cfg->x + text_offset_x + bx, in->cfg->y + (in->cfg->h - sh) / 2, sw, sh };
839
840 int content_x1 = in->cfg->x + padding;
841 int content_x2 = content_x1 + content_width;
842 if (selrect.x < content_x1) {
843 int cut = content_x1 - selrect.x;
844 selrect.x = content_x1;
845 selrect.w -= cut;
846 }
847 if (selrect.x + selrect.w > content_x2) {
848 selrect.w = content_x2 - selrect.x;
849 }
850
851 if (selrect.w > 0 && selrect.h > 0) {
852 SDL_SetRenderDrawBlendMode(renderer, SDL_BLENDMODE_BLEND);
853 SDL_SetRenderDrawColor(renderer, 51, 153, 255, 128);
854 SDL_RenderFillRect(renderer, &selrect);
855 }
856 }
857 }
858
859 SDL_Surface* surf = NULL;
860 SDL_Texture* tex = NULL;
861 /* If no text and placeholders exist, render current placeholder in gray */
862 if (in->cfg->len == 0 && in->cfg->placeholders && in->cfg->placeholder_count > 0) {
863 Uint32 now = SDL_GetTicks();
864 if (now - in->cfg->placeholder_last_tick >= 3000) { // switch placeholder every 3 seconds
865 in->cfg->placeholder_last_tick = now;
866 in->cfg->placeholder_index = (in->cfg->placeholder_index + 1) % in->cfg->placeholder_count;
867 }
868 const char* ph = in->cfg->placeholders[in->cfg->placeholder_index];
869 SDL_Color ph_color = {180, 180, 180, 255};
870 surf = TTF_RenderUTF8_Blended(font, ph, ph_color);
871 } else {
872 char saved = in->cfg->text[visible_end];
873 in->cfg->text[visible_end] = '\0';
874 surf = TTF_RenderUTF8_Blended(font, in->cfg->text + visible_start, in->cfg->text_color);
875 in->cfg->text[visible_end] = saved;
876 }
877
878 if (surf) {
879 tex = SDL_CreateTextureFromSurface(renderer, surf);
880 if (tex) {
881 int tw = 0, th = 0;
882 SDL_QueryTexture(tex, NULL, NULL, &tw, &th);
883 int dst_x = in->cfg->x + text_offset_x;
884 if (in->cfg->centered && in->cfg->len == 0 && in->cfg->placeholders && in->cfg->placeholder_count > 0) {
885 int center_off = (content_width - tw) / 2;
886 if (center_off < 0) center_off = 0;
887 dst_x = in->cfg->x + padding + center_off;
888 }
889 SDL_Rect dst = { dst_x, in->cfg->y + (in->cfg->h - th) / 2, tw, th };
890
891 SDL_Rect src = {0, 0, tw, th};
892 if (dst.w > content_width) {
893 src.w = content_width;
894 dst.w = content_width;
895 }
896
897 if (dst.w > 0 && dst.h > 0) {
898 SDL_RenderCopy(renderer, tex, &src, &dst);
899 }
900 }
901 }
902
903 /* draw cursor if focused (draw regardless of surf/tex) */
904 if (in->cfg->focused) {
905 int cursor_x = in->cfg->x + text_offset_x + cursor_x_rel;
906 int content_x1 = in->cfg->x + padding;
907 int content_x2 = content_x1 + content_width;
908 if (cursor_x < content_x1) cursor_x = content_x1;
909 if (cursor_x > content_x2) cursor_x = content_x2;
910
911 /* blink */
912 Uint32 t = SDL_GetTicks();
913 if ((t / 500) % 2 == 0) {
914 SDL_SetRenderDrawColor(renderer, in->cfg->text_color.r, in->cfg->text_color.g, in->cfg->text_color.b, in->cfg->text_color.a);
915 SDL_Rect cur = { cursor_x, in->cfg->y + 6, 2, in->cfg->h - 12 };
916 SDL_RenderFillRect(renderer, &cur);
917 }
918 }
919
920 if (tex) SDL_DestroyTexture(tex);
921 if (surf) SDL_FreeSurface(surf);
922 TTF_CloseFont(font);
923 }
924 }
