17static void input_submit_internal(AppContext* context, Input* in) {
18 if (!in || !in->cfg) return;
19
20 if (in->cfg->submit_pattern && in->cfg->text) {
21 regex_t regex;
22 if (regcomp(&regex, in->cfg->submit_pattern, REG_EXTENDED | REG_NOSUB) == 0) {
23 int invalid_submit = (regexec(&regex, in->cfg->text, 0, NULL, 0) != 0);
24 regfree(&regex);
25 if (invalid_submit) return;
26 }
27 }
28
29 in->cfg->submitted = 1;
30
31 if (in->cfg->submitted_text) {
32 free(in->cfg->submitted_text);
33 in->cfg->submitted_text = NULL;
34 }
35
36 in->cfg->submitted_text = (char*)malloc((size_t)in->cfg->len + 1);
37 if (in->cfg->submitted_text) {
38 strncpy(in->cfg->submitted_text, in->cfg->text, (size_t)in->cfg->len + 1);
39 }
40
41 const char* submitted = in->cfg->submitted_text ? in->cfg->submitted_text : in->cfg->text;
42
43 if (in->cfg->on_submit) in->cfg->on_submit(context, submitted);
44
45 if (in->cfg->text && in->cfg->maxlen > 0) {
46 in->cfg->text[0] = '\0';
47 }
48 in->cfg->len = 0;
49 in->cfg->cursor_pos = 0;
50 input_clear_selection_internal(in);
51
52 in->cfg->focused = 0;
53 SDL_StopTextInput();
54}

140Input* input_create(SDL_Renderer* renderer, InputId id, const InputConfig* cfg_in) {
141 if (input_count >= MAX_INPUTS) return NULL;
142
143 Input* in = (Input*)malloc(sizeof(Input));
144 if (!in) return NULL;
145 in->id = id;
146
147 /* Input owns its config. We copy cfg_in (or defaults) into a freshly allocated struct,
148 * then allocate runtime buffers (text) and reset pointers (textures). */
149 in->cfg = input_config_init();
150 if (!in->cfg) {
151 free(in);
152 return NULL;
153 }
154 if (cfg_in) {
155 /* Shallow copy first, then fix ownership-sensitive fields below. */
156 *in->cfg = *cfg_in;
157 }
158
159 /* runtime-managed fields: always start from a clean state */
160 in->cfg->bg_texture = NULL;
161 in->cfg->submitted_text = NULL;
162 in->cfg->text = NULL;
163 in->cfg->len = 0;
164 in->cfg->cursor_pos = 0;
165 in->cfg->focused = 0;
166 in->cfg->submitted = 0;
167 in->cfg->sel_start = 0;
168 in->cfg->sel_len = 0;
169 in->cfg->sel_anchor = 0;
170 in->cfg->placeholder_index = 0;
171 in->cfg->placeholder_last_tick = SDL_GetTicks();
172
173 /* Conversion coordonnées relatives au centre -> écran.
174 * x positif = droite, y positif = haut.
175 * (x, y) représente le centre de l'input relatif au centre de la fenêtre. */
176 int rel_x = in->cfg->x;
177 int rel_y = in->cfg->y;
178 int screen_x = (WIN_WIDTH / 2) + rel_x - (in->cfg->w / 2);
179 int screen_y = (WIN_HEIGHT / 2) - rel_y - (in->cfg->h / 2);
180
181 in->cfg->x = screen_x;
182 in->cfg->y = screen_y;
183
184 /* keep rect in sync */
185 in->cfg->rect = (SDL_Rect){in->cfg->x, in->cfg->y, in->cfg->w, in->cfg->h};
186
187 /* allocate text buffer */
188 if (in->cfg->maxlen <= 0) in->cfg->maxlen = INPUT_DEFAULT_MAX;
189 in->cfg->text = (char*)calloc((size_t)in->cfg->maxlen + 1, 1);
190 if (!in->cfg->text) {
191 input_destroy(in);
192 return NULL;
193 }
194
195 /* placeholders: just reference caller-provided strings (do not strdup),
196 * so stack/const arrays are safe as long as they outlive the input.
197 * (This matches current usage in menu.c.) */
198 /* submitted_label is a const char*; no allocation here */
199
200 if (in->cfg->init_text && in->cfg->init_text[0] != '\0') {
201 strncpy(in->cfg->text, in->cfg->init_text, (size_t)in->cfg->maxlen);
202 in->cfg->text[in->cfg->maxlen] = '\0';
203 in->cfg->len = (int)strlen(in->cfg->text);
204 in->cfg->cursor_pos = in->cfg->len;
205 }
206
207 /* Load saved player name from properties file if persistence is enabled */
208 if (in->cfg->save_player_data && in->id == INPUT_ID_NAME) {
209 char saved_name[INPUT_DEFAULT_MAX + 1] = {0};
210 if (read_property(saved_name, "PLAYER_NAME") == EXIT_SUCCESS && saved_name[0] != '\0') {
211 strncpy(in->cfg->text, saved_name, (size_t)in->cfg->maxlen);
212 in->cfg->text[in->cfg->maxlen] = '\0';
213 in->cfg->len = (int)strlen(in->cfg->text);
214 in->cfg->cursor_pos = in->cfg->len;
215 }
216 }
217
218 /* auto-load background image if bg_path was provided in config */
219 if (in->cfg->bg_path && renderer) {
220 input_set_bg(in, renderer, in->cfg->bg_path, in->cfg->bg_padding);
221 }
222
223 inputs[input_count++] = in;
224
225 return in;
226}

228void input_destroy(Input* in) {
229 if (!in) return;
230
231 for (int i = 0; i < input_count; ++i) {
232 if (inputs[i] == in) {
233 for (int j = i; j < input_count - 1; ++j) inputs[j] = inputs[j + 1];
234 inputs[input_count - 1] = NULL;
235 input_count--;
236 break;
237 }
238 }
239
240 if (in->cfg) {
241 if (in->cfg->text) free(in->cfg->text);
242 if (in->cfg->bg_texture) free_image(in->cfg->bg_texture);
243 if (in->cfg->submitted_text) free(in->cfg->submitted_text);
244 /* submitted_label and placeholders are not owned (const pointers) */
245 free(in->cfg);
246 }
247 free(in);
248}

252static int input_text_allowed(const char* allowed_pattern, const char* txt) {
253 if (!allowed_pattern || !txt) return 1;
254 regex_t regex;
255 if (regcomp(&regex, allowed_pattern, REG_EXTENDED | REG_NOSUB) != 0) return 1; /* regex invalide → tout accepter */
256 int len = (int)strlen(txt);
257 /* tester chaque caractère individuellement */
258 for (int i = 0; i < len; i++) {
259 char ch[2] = { txt[i], '\0' };
260 if (regexec(&regex, ch, 0, NULL, 0) != 0) {
261 regfree(&regex);
262 return 0; /* caractère refusé */
263 }
264 }
265 regfree(&regex);
266 return 1;
267}

273void input_handle_event(AppContext* context, Input* in, SDL_Event* e) {
274 if (!in || !e) return;
275
276 if (e->type == SDL_MOUSEBUTTONDOWN) {
277 int mx = e->button.x;
278 int my = e->button.y;
279 if (point_in_rect(mx, my, &in->cfg->rect)) {
280 in->cfg->focused = 1;
281 SDL_StartTextInput();
282 /* place cursor at end */
283 in->cfg->cursor_pos = in->cfg->len;
284 } else {
285 if (in->cfg->focused) {
286 in->cfg->focused = 0;
287 SDL_StopTextInput();
288 }
289 }
290 }
291
292 if (!in->cfg->focused) return;
293
294 if (e->type == SDL_TEXTINPUT) {
295 const char* txt = e->text.text;
296 int add = strlen(txt);
297 if (in->cfg->len + add > in->cfg->maxlen) return;
298 /* filter characters against allowed_pattern if set */
299 if (!input_text_allowed(in->cfg->allowed_pattern, txt)) return;
300 /* if there is a selection, replace it */
301 if (input_has_selection(in)) {
302 input_delete_selection(in);
303 }
304 /* insert at cursor_pos */
305 memmove(in->cfg->text + in->cfg->cursor_pos + add, in->cfg->text + in->cfg->cursor_pos, in->cfg->len - in->cfg->cursor_pos + 1);
306 memcpy(in->cfg->text + in->cfg->cursor_pos, txt, add);
307 in->cfg->cursor_pos += add;
308 in->cfg->len += add;
309 }
310
311 if (e->type == SDL_KEYDOWN) {
312 SDL_Keycode k = e->key.keysym.sym;
313 SDL_Keymod mod = e->key.keysym.mod;
314 /* Ctrl+A = select all */
315 if ((mod & KMOD_CTRL) && (k == SDLK_a)) {
316 in->cfg->sel_start = 0;
317 in->cfg->sel_len = in->cfg->len;
318 in->cfg->sel_anchor = 0;
319 in->cfg->cursor_pos = in->cfg->len;
320 return;
321 }
322
323 if (k == SDLK_BACKSPACE) {
324 if (input_has_selection(in)) {
325 input_delete_selection(in);
326 } else if (mod & KMOD_CTRL) {
327 int np = prev_word_pos(in, in->cfg->cursor_pos);
328 int del = in->cfg->cursor_pos - np;
329 if (del > 0) {
330 memmove(in->cfg->text + np, in->cfg->text + in->cfg->cursor_pos, in->cfg->len - in->cfg->cursor_pos + 1);
331 in->cfg->len -= del;
332 in->cfg->cursor_pos = np;
333 in->cfg->text[in->cfg->len] = '\0';
334 }
335 } else {
336 if (in->cfg->cursor_pos > 0 && in->cfg->len > 0) {
337 memmove(in->cfg->text + in->cfg->cursor_pos - 1, in->cfg->text + in->cfg->cursor_pos, in->cfg->len - in->cfg->cursor_pos + 1);
338 in->cfg->cursor_pos--;
339 in->cfg->len--;
340 }
341 }
342 input_clear_selection_internal(in);
343 } else if (k == SDLK_DELETE) {
344 if (input_has_selection(in)) {
345 input_delete_selection(in);
346 } else if (mod & KMOD_CTRL) {
347 int np = next_word_pos(in, in->cfg->cursor_pos);
348 int del = np - in->cfg->cursor_pos;
349 if (del > 0) {
350 memmove(in->cfg->text + in->cfg->cursor_pos, in->cfg->text + np, in->cfg->len - np + 1);
351 in->cfg->len -= del;
352 in->cfg->text[in->cfg->len] = '\0';
353 }
354 } else {
355 if (in->cfg->cursor_pos < in->cfg->len && in->cfg->len > 0) {
356 memmove(in->cfg->text + in->cfg->cursor_pos, in->cfg->text + in->cfg->cursor_pos + 1, in->cfg->len - in->cfg->cursor_pos);
357 in->cfg->len--;
358 in->cfg->text[in->cfg->len] = '\0';
359 }
360 }
361 input_clear_selection_internal(in);
362 } else if (k == SDLK_LEFT) {
363 int newpos = in->cfg->cursor_pos;
364 if (mod & KMOD_CTRL) newpos = prev_word_pos(in, in->cfg->cursor_pos);
365 else if (in->cfg->cursor_pos > 0) newpos = in->cfg->cursor_pos - 1;
366
367 if (mod & KMOD_SHIFT) {
368 if (!input_has_selection(in)) {
369 in->cfg->sel_anchor = in->cfg->cursor_pos;
370 in->cfg->sel_start = in->cfg->cursor_pos;
371 }
372 in->cfg->cursor_pos = newpos;
373 int s = in->cfg->sel_anchor;
374 int epos = in->cfg->cursor_pos;
375 if (epos < s) { in->cfg->sel_start = epos; in->cfg->sel_len = s - epos; }
376 else { in->cfg->sel_start = s; in->cfg->sel_len = epos - s; }
377 } else {
378 in->cfg->cursor_pos = newpos;
379 input_clear_selection_internal(in);
380 }
381 } else if (k == SDLK_RIGHT) {
382 int newpos = in->cfg->cursor_pos;
383 if (mod & KMOD_CTRL) newpos = next_word_pos(in, in->cfg->cursor_pos);
384 else if (in->cfg->cursor_pos < in->cfg->len) newpos = in->cfg->cursor_pos + 1;
385
386 if (mod & KMOD_SHIFT) {
387 if (!input_has_selection(in)) {
388 in->cfg->sel_anchor = in->cfg->cursor_pos;
389 in->cfg->sel_start = in->cfg->cursor_pos;
390 }
391 in->cfg->cursor_pos = newpos;
392 int s = in->cfg->sel_anchor;
393 int epos = in->cfg->cursor_pos;
394 if (epos < s) { in->cfg->sel_start = epos; in->cfg->sel_len = s - epos; }
395 else { in->cfg->sel_start = s; in->cfg->sel_len = epos - s; }
396 } else {
397 in->cfg->cursor_pos = newpos;
398 input_clear_selection_internal(in);
399 }
400 } else if (k == SDLK_RETURN || k == SDLK_KP_ENTER) {
401 input_submit_internal(context, in);
402 }
403 }

406void input_render(SDL_Renderer* renderer, Input* in) {
407 if (!renderer || !in) return;
408 SDL_Rect rect = (SDL_Rect){ in->cfg->x, in->cfg->y, in->cfg->w, in->cfg->h };
409 in->cfg->rect = rect;
410
411 /* background: image if present else color */
412 if (in->cfg->bg_texture) {
413 SDL_RenderCopy(renderer, in->cfg->bg_texture, NULL, &rect);
414 } else {
415 SDL_SetRenderDrawColor(renderer, in->cfg->bg_color.r, in->cfg->bg_color.g, in->cfg->bg_color.b, in->cfg->bg_color.a);
416 SDL_RenderFillRect(renderer, &rect);
417 }
418
419 /* border: draw only when no background texture is set */
420 if (!in->cfg->bg_texture) {
421 SDL_SetRenderDrawColor(renderer, in->cfg->border_color.r, in->cfg->border_color.g, in->cfg->border_color.b, in->cfg->border_color.a);
422 SDL_RenderDrawRect(renderer, &rect);
423 }
424
425 /* render submitted label (always if set) and submitted text (if present) above input */
426 if (in->cfg->font_path && in->cfg->submitted_label) {
427 int label_size = in->cfg->font_size > 12 ? in->cfg->font_size - 4 : in->cfg->font_size;
428 TTF_Font* label_font = TTF_OpenFont(in->cfg->font_path, label_size);
429 if (label_font) {
430 SDL_Color color = {255,255,255,255};
431 const char* lbl = in->cfg->submitted_label ? in->cfg->submitted_label : "";
432 const char* sub = in->cfg->submitted_text ? in->cfg->submitted_text : "";
433 size_t L = strlen(lbl);
434 size_t M = strlen(sub);
435 char* combined = (char*)malloc(L + M + 1);
436 if (combined) {
437 memcpy(combined, lbl, L);
438 memcpy(combined + L, sub, M + 1);
439 SDL_Surface* surf_label = TTF_RenderUTF8_Blended(label_font, combined, color);
440 if (surf_label) {
441 SDL_Texture* tex_label = SDL_CreateTextureFromSurface(renderer, surf_label);
442 if (tex_label) {
443 int tw = 0, th = 0;
444 SDL_QueryTexture(tex_label, NULL, NULL, &tw, &th);
445 int dst_x = in->cfg->x + in->cfg->padding;
446 int dst_y = in->cfg->y - th - 8;
447 SDL_Rect dst = { dst_x, dst_y, tw, th };
448 SDL_RenderCopy(renderer, tex_label, NULL, &dst);
449 SDL_DestroyTexture(tex_label);
450 }
451 SDL_FreeSurface(surf_label);
452 }
453 free(combined);
454 }
455 TTF_CloseFont(label_font);
456 }
457 }
458
459 /* render text using TTF */
460 if (in->cfg->font_path && in->cfg->font_size > 0) {
461 TTF_Font* font = TTF_OpenFont(in->cfg->font_path, in->cfg->font_size);
462 if (font) {
463 /* compute width of text up to cursor so we can draw cursor even when no
464 * texture is created (empty string case) */
465 char before_cursor[512];
466 int n = in->cfg->cursor_pos;
467 if (n > 0) {
468 if (n >= (int)sizeof(before_cursor)) n = sizeof(before_cursor)-1;
469 memcpy(before_cursor, in->cfg->text, n);
470 before_cursor[n] = '\0';
471 } else before_cursor[0] = '\0';
472
473 int cx = 0, cy = 0;
474 TTF_SizeUTF8(font, before_cursor, &cx, &cy);
475 int padding = in->cfg->padding;
476
477 /* compute full text width to determine centering offset */
478 int full_tw = 0, full_th = 0;
479 TTF_SizeUTF8(font, in->cfg->text, &full_tw, &full_th);
480 int text_offset_x = padding; /* default: left-aligned with padding */
481 if (in->cfg->centered) {
482 int center_off = (in->cfg->w - full_tw) / 2;
483 if (center_off < padding) center_off = padding;
484 text_offset_x = center_off;
485 }
486
487 /* draw selection highlight if any */
488 if (input_has_selection(in)) {
489 int s = in->cfg->sel_start;
490 int l = in->cfg->sel_len;
491 if (s < 0) s = 0;
492 if (s > in->cfg->len) s = in->cfg->len;
493 if (l < 0) l = 0;
494 if (s + l > in->cfg->len) l = in->cfg->len - s;
495 if (l > 0) {
496 char* before_sel = (char*)malloc(s + 1);
497 char* sel_text = (char*)malloc(l + 1);
498 if (before_sel && sel_text) {
499 if (s > 0) memcpy(before_sel, in->cfg->text, s);
500 before_sel[s] = '\0';
501 memcpy(sel_text, in->cfg->text + s, l);
502 sel_text[l] = '\0';
503 int bx = 0, by = 0, sw = 0, sh = 0;
504 TTF_SizeUTF8(font, before_sel, &bx, &by);
505 TTF_SizeUTF8(font, sel_text, &sw, &sh);
506 SDL_Rect selrect = { in->cfg->x + text_offset_x + bx, in->cfg->y + (in->cfg->h - sh) / 2, sw, sh };
507 if (selrect.x < in->cfg->x + padding) selrect.x = in->cfg->x + padding;
508 if (selrect.x + selrect.w > in->cfg->x + in->cfg->w - padding) selrect.w = (in->cfg->x + in->cfg->w - padding) - selrect.x;
509 SDL_SetRenderDrawBlendMode(renderer, SDL_BLENDMODE_BLEND);
510 SDL_SetRenderDrawColor(renderer, 51, 153, 255, 128);
511 SDL_RenderFillRect(renderer, &selrect);
512 }
513 free(before_sel);
514 free(sel_text);
515 }
516 }
517
518 SDL_Surface* surf = NULL;
519 SDL_Texture* tex = NULL;
520 /* If no text and placeholders exist, render current placeholder in gray */
521 if (in->cfg->len == 0 && in->cfg->placeholders && in->cfg->placeholder_count > 0) {
522 Uint32 now = SDL_GetTicks();
523 if (now - in->cfg->placeholder_last_tick >= 3000) { // switch placeholder every 3 seconds
524 in->cfg->placeholder_last_tick = now;
525 in->cfg->placeholder_index = (in->cfg->placeholder_index + 1) % in->cfg->placeholder_count;
526 }
527 const char* ph = in->cfg->placeholders[in->cfg->placeholder_index];
528 SDL_Color ph_color = {180, 180, 180, 255};
529 surf = TTF_RenderUTF8_Blended(font, ph, ph_color);
530 } else {
531 surf = TTF_RenderUTF8_Blended(font, in->cfg->text, in->cfg->text_color);
532 }
533 if (surf) {
534 tex = SDL_CreateTextureFromSurface(renderer, surf);
535 if (tex) {
536 int tw = 0, th = 0;
537 SDL_QueryTexture(tex, NULL, NULL, &tw, &th);
538 /* For placeholder centering, recompute offset based on placeholder width */
539 int dst_x = in->cfg->x + text_offset_x;
540 if (in->cfg->centered && in->cfg->len == 0 && in->cfg->placeholders && in->cfg->placeholder_count > 0) {
541 int center_off = (in->cfg->w - tw) / 2;
542 if (center_off < padding) center_off = padding;
543 dst_x = in->cfg->x + center_off;
544 }
545 SDL_Rect dst = { dst_x, in->cfg->y + (in->cfg->h - th) / 2, tw, th };
546
547 /* If text wider than input, clip source width */
548 SDL_Rect src = {0, 0, tw, th};
549 if (dst.w > in->cfg->w - padding*2) {
550 src.w = (in->cfg->w - padding*2) * ((float)tw / dst.w);
551 dst.w = in->cfg->w - padding*2;
552 }
553
554 SDL_RenderCopy(renderer, tex, &src, &dst);
555 }
556 }
557
558 /* draw cursor if focused (draw regardless of surf/tex) */
559 if (in->cfg->focused) {
560 int cursor_x = in->cfg->x + text_offset_x + cx;
561 /* ensure cursor stays inside rect */
562 if (cursor_x > in->cfg->x + in->cfg->w - padding) cursor_x = in->cfg->x + in->cfg->w - padding;
563
564 /* blink */
565 Uint32 t = SDL_GetTicks();
566 if ((t / 500) % 2 == 0) {
567 SDL_SetRenderDrawColor(renderer, in->cfg->text_color.r, in->cfg->text_color.g, in->cfg->text_color.b, in->cfg->text_color.a);
568 SDL_Rect cur = { cursor_x, in->cfg->y + 6, 2, in->cfg->h - 12 };
569 SDL_RenderFillRect(renderer, &cur);
570 }
571 }
572
573 if (tex) SDL_DestroyTexture(tex);
574 if (surf) SDL_FreeSurface(surf);
575 TTF_CloseFont(font);
576 }
577 }

614int input_set_bg(Input* in, SDL_Renderer* renderer, const char* path, int padding) {
615 if (!in || !renderer || !path) return EXIT_FAILURE;
616 if (in->cfg->bg_texture) {
617 free_image(in->cfg->bg_texture);
618 in->cfg->bg_texture = NULL;
619 }
620 SDL_Texture* tex = load_image(renderer, path);
621 if (!tex) return EXIT_FAILURE;
622 /* If caller provided padding >= 0, use it; otherwise adapt to texture */
623 if (padding >= 0) {
624 in->cfg->padding = padding;
625 } else {
626 int tw = 0, th = 0;
627 if (SDL_QueryTexture(tex, NULL, NULL, &tw, &th) == 0) {
628 int p = in->cfg->h / 3;
629 if (p < 6) p = 6;
630 if (tw > 0 && tw < in->cfg->w) p += 4;
631 in->cfg->padding = p;
632 }
633 }
634 in->cfg->bg_texture = tex;
635 return EXIT_SUCCESS;
636}

653int edit_in_cfg(InputId id, InputCfgKey key, intptr_t value) {
654 Input* in = input_find_by_id(id);

656
657 switch (key) {

675 in->cfg->font_path = (const char*)value;
676 return EXIT_SUCCESS;
677 case IN_CFG_FONT_SIZE:
678 in->cfg->font_size = (int)value;
679 return EXIT_SUCCESS;
680 case IN_CFG_PLACEHOLDERS:
681 in->cfg->placeholders = (const char**)value;
682 return EXIT_SUCCESS;
683 case IN_CFG_PLACEHOLDER_COUNT:
684 in->cfg->placeholder_count = (int)value;
685 return EXIT_SUCCESS;
686 case IN_CFG_SUBMITTED_LABEL:
687 in->cfg->submitted_label = (const char*)value;
688 return EXIT_SUCCESS;
689 case IN_CFG_MAXLEN: {
690 int new_maxlen = (int)value;
691 if (new_maxlen <= 0) return EXIT_FAILURE;
692
693 char* new_text = (char*)calloc((size_t)new_maxlen + 1, 1);
694 if (!new_text) return EXIT_FAILURE;
695
696 if (in->cfg->text) {
697 size_t old_len = strlen(in->cfg->text);
698 size_t copy_len = old_len;
699 if (copy_len > (size_t)new_maxlen) copy_len = (size_t)new_maxlen;
700 memcpy(new_text, in->cfg->text, copy_len);
701 new_text[copy_len] = '\0';
702 free(in->cfg->text);
703 }
704
705 in->cfg->text = new_text;
706 in->cfg->maxlen = new_maxlen;
707 in->cfg->len = (int)strlen(in->cfg->text);
708 if (in->cfg->cursor_pos > in->cfg->len) in->cfg->cursor_pos = in->cfg->len;
709 if (in->cfg->sel_start > in->cfg->len) in->cfg->sel_start = in->cfg->len;
710 if (in->cfg->sel_start + in->cfg->sel_len > in->cfg->len) {
711 in->cfg->sel_len = in->cfg->len - in->cfg->sel_start;
712 if (in->cfg->sel_len < 0) in->cfg->sel_len = 0;
713 }
714 return EXIT_SUCCESS;
715 }
716 case IN_CFG_SAVE_PLAYER_DATA:
717 in->cfg->save_player_data = ((int)value != 0);
718 return EXIT_SUCCESS;
719 case IN_CFG_BG_COLOR:
720 if (!value) return EXIT_FAILURE;
721 in->cfg->bg_color = *(const SDL_Color*)value;
722 return EXIT_SUCCESS;
723 case IN_CFG_BORDER_COLOR:

725 in->cfg->border_color = *(const SDL_Color*)value;
726 return EXIT_SUCCESS;
727 case IN_CFG_TEXT_COLOR:

740 return EXIT_SUCCESS;

751 in->cfg->init_text = (const char*)value;
752 return EXIT_SUCCESS;
753 case IN_CFG_RECT:
754 if (!value) return EXIT_FAILURE;
755 in->cfg->rect = *(const SDL_Rect*)value;
756 in->cfg->x = in->cfg->rect.x;
757 in->cfg->y = in->cfg->rect.y;
758 in->cfg->w = in->cfg->rect.w;
759 in->cfg->h = in->cfg->rect.h;
760 return EXIT_SUCCESS;
761 case IN_CFG_TEXT: {
762 const char* text_value = (const char*)value;
763 if (!text_value) text_value = "";
764 input_set_text(in, text_value);
765 return EXIT_SUCCESS;
766 }
767 case IN_CFG_LEN: {
768 int new_len = (int)value;
769 if (new_len < 0) new_len = 0;
770 int current_len = in->cfg->text ? (int)strlen(in->cfg->text) : 0;
771 if (new_len > current_len) new_len = current_len;
772 in->cfg->len = new_len;
773 if (in->cfg->text) in->cfg->text[new_len] = '\0';
774 if (in->cfg->cursor_pos > in->cfg->len) in->cfg->cursor_pos = in->cfg->len;
775 return EXIT_SUCCESS;
776 }
777 case IN_CFG_CURSOR_POS:
778 in->cfg->cursor_pos = (int)value;
779 if (in->cfg->cursor_pos < 0) in->cfg->cursor_pos = 0;
780 if (in->cfg->cursor_pos > in->cfg->len) in->cfg->cursor_pos = in->cfg->len;
781 return EXIT_SUCCESS;
782 case IN_CFG_FOCUSED:
783 in->cfg->focused = ((int)value != 0);
784 if (in->cfg->focused) SDL_StartTextInput();
785 else SDL_StopTextInput();
786 return EXIT_SUCCESS;
787 case IN_CFG_SUBMITTED:
788 in->cfg->submitted = ((int)value != 0);
789 return EXIT_SUCCESS;
790 case IN_CFG_BG_TEXTURE:
791 in->cfg->bg_texture = (SDL_Texture*)value;
792 return EXIT_SUCCESS;
793 case IN_CFG_SEL_START:
794 in->cfg->sel_start = (int)value;
795 return EXIT_SUCCESS;

797 in->cfg->sel_len = (int)value;
798 return EXIT_SUCCESS;
799 case IN_CFG_SEL_ANCHOR:
800 in->cfg->sel_anchor = (int)value;
801 return EXIT_SUCCESS;
802 case IN_CFG_SUBMITTED_TEXT: {
803 const char* submitted_value = (const char*)value;
804 if (in->cfg->submitted_text) {
805 free(in->cfg->submitted_text);
806 in->cfg->submitted_text = NULL;
807 }
808 if (submitted_value) {

810 if (!in->cfg->submitted_text) return EXIT_FAILURE;
811 }
812 return EXIT_SUCCESS;

822 in->cfg->on_submit = *(void (**)(AppContext*, const char*))value;
823 return EXIT_SUCCESS;
824 default:
825 return EXIT_FAILURE;
826 }
