842static int compute_panel_member_position(int nb_player, int i_player, int base_x, int base_y, int* out_x, int* out_y) {
843 if (!out_x || !out_y) return 0;
844 if (nb_player < 1 || nb_player > 8) return 0;
845 if (i_player < 0 || i_player >= nb_player) return 0;
846
847 const int spacing_x = 60;
848 const int spacing_y = 64;
849
850 int first_line_count = 0;
851 switch (nb_player) {
852 case 1: first_line_count = 1; break;
853 case 2: first_line_count = 2; break;
854 case 3: first_line_count = 3; break;
855 case 4: first_line_count = 4; break;
856 case 5: first_line_count = 3; break;
857 case 6: first_line_count = 3; break;
858 case 7: first_line_count = 4; break;
859 case 8: first_line_count = 4; break;
860 default: return 0;
861 }
862
863 int second_line_count = nb_player - first_line_count;
864 int is_second_line = (i_player >= first_line_count);
865 int line_count = is_second_line ? second_line_count : first_line_count;
866 int line_index = is_second_line ? (i_player - first_line_count) : i_player;
867
868 *out_y = is_second_line ? (base_y - spacing_y) : base_y;
869
870 switch (line_count) {
871 case 1:
872 *out_x = base_x;
873 break;
874 case 2:
875 if (is_second_line) {
876 *out_x = base_x + ((line_index == 0) ? -spacing_x/2 : spacing_x/2); // Pour les 2 joueurs de la deuxième ligne : cas avec 5 joueurs.
877 } else {
878 *out_x = base_x + ((line_index == 0) ? -spacing_x : spacing_x);
879 }
880 break;
881 case 3:
882 *out_x = base_x + (line_index - 1) * spacing_x;
883 break;
884 case 4:
885 *out_x = base_x + (line_index * spacing_x) - (int)(1.5f * spacing_x);
886 break;
887 default:
888 return 0;
889 }
890
891 return 1;
892}
