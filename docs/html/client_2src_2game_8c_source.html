413static void render_team_member_in_panel(AppContext* context, Window* panel, SDL_Texture* icon, Text* txt, const char* name, UserRole role, int* spy_row, int* agent_row) {
414 if (!context || !panel || !icon || !txt || !spy_row || !agent_row) return;
415
416 const int POS_X = 0;
417 const int AGENT_BASE_Y = 65;
418 const int SPY_BASE_Y = -85; // Ecart de 150 entre la position de l'espion et des agents
419 const int ROW_GAP = 36;
420
421 int row = (role == ROLE_SPY) ? *spy_row : *agent_row;
422 int icon_y = ((role == ROLE_SPY) ? SPY_BASE_Y : AGENT_BASE_Y) + (row * ROW_GAP);
423
424 window_display_image(context->renderer, panel, icon, POS_X, icon_y+25, 0.20f, 0, SDL_FLIP_NONE, 1, 255);
425
426 update_text(context, txt, name ? name : "???");
427 window_place_text(panel, txt, POS_X, icon_y);
428 display_text(context, txt);
429
430 if (role == ROLE_SPY) (*spy_row)++;
431 else (*agent_row)++;
432}

434static void render_team_panel_content(AppContext* context, Window* panel, Team team, SDL_Texture* icon, Text* txt_spy_label, Text* txt_agents_label, Text** player_texts, int* player_index) {
435 if (!context || !panel || !txt_spy_label || !txt_agents_label || !player_texts || !player_index) return;
436
437 int spy_row = 0;
438 int agent_row = 0;
439
440 update_text(context, txt_spy_label, "Espions :");
441 window_place_text(panel, txt_spy_label, 0, -25);
442 display_text(context, txt_spy_label);
443
444 update_text(context, txt_agents_label, "Agents :");
445 window_place_text(panel, txt_agents_label, 0, 125); // Ecart de 150 entre la position de l'espion et des agents
446 display_text(context, txt_agents_label);
447
448 *player_index = 0;
449
450 if (!context->lobby) return;
451
452 if (context->player_team == team && *player_index < MAX_TEAM_PLAYERS) {
453 Text* txt = player_texts[*player_index];
454 const char* local_name = context->player_name ? context->player_name : "Moi";
455 // Les membres sont placés en fonctions des étiquettes des rôles
456 render_team_member_in_panel(context, panel, icon, txt, local_name, context->player_role, &spy_row, &agent_row);
457 (*player_index)++;
458 }
459
460 for (int i = 0; i < context->lobby->nb_players && i < MAX_USERS; i++) {
461 User* u = context->lobby->users[i];
462 if (!u || u->team != team || *player_index >= MAX_TEAM_PLAYERS) continue;
463
464 Text* txt = player_texts[*player_index];
465 // Les membres sont placés en fonctions des étiquettes des rôles
466 render_team_member_in_panel(context, panel, icon, txt, u->name ? u->name : "???", u->role, &spy_row, &agent_row);
467 (*player_index)++;
468 }
469}

482void game_render_cards(AppContext * context) {
483 int x=-400;
484 int y=-250;
485 int card_index = 0;
486 for (int i = 0; i < 5; i++) {
487 for (int j = 0; j < 5; j++) {
488 Word word = context->lobby->game->words[i*5 + j];
489
490 if(context->player_role == ROLE_SPY && !word.revealed) {
491 switch (word.team) {
492 case TEAM_NONE:
493 if (word.gender) {
494 display_image(context->renderer, card_f_classic, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
495 }else{
496 display_image(context->renderer, card_h_classic, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
497 }
498 break;
499 case TEAM_RED:
500 if (word.gender) {
501 display_image(context->renderer, card_f_red, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
502 }else{
503 display_image(context->renderer, card_h_red, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
504 }
505 break;
506 case TEAM_BLUE:
507 if (word.gender) {
508 display_image(context->renderer, card_f_blue, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
509 }else{
510 display_image(context->renderer, card_h_blue, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
511 }
