316void game_render_cards(AppContext * context) {
317 int x=-400;
318 int y=-250;
319 int card_index = 0;
320 for (int i = 0; i < 5; i++) {
321 for (int j = 0; j < 5; j++) {
322 Word word = context->lobby->game->words[i*5 + j];
323
324 if(context->player_role == ROLE_SPY && !word.revealed) {
325 switch (word.team) {
326 case TEAM_NONE:
327 if (word.gender) {
328 display_image(context->renderer, card_f_classic, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
329 }else{
330 display_image(context->renderer, card_h_classic, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
331 }
332 break;
333 case TEAM_RED:
334 if (word.gender) {
335 display_image(context->renderer, card_f_red, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
336 }else{
337 display_image(context->renderer, card_h_red, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
338 }
339 break;
340 case TEAM_BLUE:
341 if (word.gender) {
342 display_image(context->renderer, card_f_blue, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
343 }else{
344 display_image(context->renderer, card_h_blue, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
345 }
346 break;
347 case TEAM_BLACK:
348 display_image(context->renderer, card_black, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
349 default:
350 break;
351 }
352 }else if (context->player_role == ROLE_AGENT && !word.revealed) {
353 if (word.gender) {
354 display_image(context->renderer, card_f_classic, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
355 }else{
356 display_image(context->renderer, card_h_classic, x, y, 0.04, 0, SDL_FLIP_NONE, 1, 255);
357 }
358 }
359 else if (word.revealed) {
360 // Attente des images des cartes "révélées"
361 }
362
363 /* Affichage du mot de la carte avec le nouveau système */
364 if (card_index < NUM_CARDS && txt_card_words[card_index]) {
365 update_text(context, txt_card_words[card_index], word.word);
366 update_text_position(txt_card_words[card_index], x, y - 22);
367 display_text(context, txt_card_words[card_index]);
368 }
369 card_index++;
370 x += 200;
371 }
372 x = -400;
373 y += 125;
374 }
