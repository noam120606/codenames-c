82int levenshtein(const char* s1, const char* s2) {
83 if (!s1 || !s2) return -1;
84
85 size_t len1 = strlen(s1);
86 size_t len2 = strlen(s2);
87
88 // Cas triviaux
89 if (len1 == 0) return (int)len2;
90 if (len2 == 0) return (int)len1;
91
92 // Allouer la matrice de distances
93 int** matrix = malloc((len1 + 1) * sizeof(int*));
94 if (!matrix) return -1;
95
96 for (size_t i = 0; i <= len1; i++) {
97 matrix[i] = malloc((len2 + 1) * sizeof(int));
98 if (!matrix[i]) {
99 for (size_t j = 0; j < i; j++) free(matrix[j]);
100 free(matrix);
101 return -1;
102 }
103 }
104
105 // Initialiser la première colonne et ligne
106 for (size_t i = 0; i <= len1; i++) matrix[i][0] = (int)i;
107 for (size_t j = 0; j <= len2; j++) matrix[0][j] = (int)j;
108
109 // Remplir la matrice
110 for (size_t i = 1; i <= len1; i++) {
111 for (size_t j = 1; j <= len2; j++) {
112 // Comparaison insensible à la casse
113 int cost = (tolower((unsigned char)s1[i-1]) == tolower((unsigned char)s2[j-1])) ? 0 : 1;
114
115 int del = matrix[i-1][j] + 1; // Suppression
116 int ins = matrix[i][j-1] + 1; // Insertion
117 int sub = matrix[i-1][j-1] + cost; // Substitution
118
119 // Minimum des trois opérations
120 int min = del;
121 if (ins < min) min = ins;
122 if (sub < min) min = sub;
123
124 matrix[i][j] = min;
125 }
126 }
127
128 int result = matrix[len1][len2];
129
130 // Libérer la mémoire
131 for (size_t i = 0; i <= len1; i++) free(matrix[i]);
132 free(matrix);
133
134 return result;
135}
