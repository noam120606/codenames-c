110int levenshtein(const char* s1, const char* s2) {
111 if (!s1 || !s2) return -1;
112
113 size_t len1 = strlen(s1);
114 size_t len2 = strlen(s2);
115
116 // Cas triviaux
117 if (len1 == 0) return (int)len2;
118 if (len2 == 0) return (int)len1;
119
120 // Allouer la matrice de distances
121 int** matrix = malloc((len1 + 1) * sizeof(int*));
122 if (!matrix) return -1;
123
124 for (size_t i = 0; i <= len1; i++) {
125 matrix[i] = malloc((len2 + 1) * sizeof(int));
126 if (!matrix[i]) {
127 for (size_t j = 0; j < i; j++) free(matrix[j]);
128 free(matrix);
129 return -1;
130 }
131 }
132
133 // Initialiser la première colonne et ligne
134 for (size_t i = 0; i <= len1; i++) matrix[i][0] = (int)i;
135 for (size_t j = 0; j <= len2; j++) matrix[0][j] = (int)j;
136
137 // Remplir la matrice
138 for (size_t i = 1; i <= len1; i++) {
139 for (size_t j = 1; j <= len2; j++) {
140 // Comparaison insensible à la casse
141 int cost = (tolower((unsigned char)s1[i-1]) == tolower((unsigned char)s2[j-1])) ? 0 : 1;
142
143 int del = matrix[i-1][j] + 1; // Suppression
144 int ins = matrix[i][j-1] + 1; // Insertion
145 int sub = matrix[i-1][j-1] + cost; // Substitution
146
147 // Minimum des trois opérations
148 int min = del;
149 if (ins < min) min = ins;
150 if (sub < min) min = sub;
151
152 matrix[i][j] = min;
153 }
154 }
155
156 int result = matrix[len1][len2];
157
158 // Libérer la mémoire
159 for (size_t i = 0; i <= len1; i++) free(matrix[i]);
160 free(matrix);
161
162 return result;
163}
